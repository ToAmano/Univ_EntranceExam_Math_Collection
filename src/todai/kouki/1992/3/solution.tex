  {\bf[解]}

  (1)
  $n$に関する数学的帰納法で題意($\diamondsuit$とおく)を示す．

  \vspace{5pt}
  \textbf{$1^{\circ}$ $n=1$の時}

  $n=1$の時 $P_1(x)=1$であり，$m=1$ならばあまりは$P_0(x)=0$である．
  また$m \ge 2$の時はあまりは$P_{1}(x)$自身である．
  よって$\diamondsuit$は成立する．

  \vspace{5pt}
  \textbf{$1^{\circ}$ $n\le k$での成立を仮定}

  $n \le k$での$\diamondsuit$の成立を仮定して，$n=k+1$での成立を示す．
  仮定から，適当な多項式$R(x)$があって，
  \begin{align}
    P_k(x) & = R_i(x) \cdot P_m(x) + P_i(x) \quad (i=0,1,\dots,m-1) \label{1992-3:eq:1}
  \end{align}
  と書ける．
  $P_{k+1}(x)=x P_k(x)+1$だから，\cref{1992-3:eq:1}を代入して
  \begin{align}
    P_{k+1}(x)
     & = x \left(R_i(x) \cdot P_m(x) + P_i(x) \right) + 1               \\
     & = x \cdot R_i(x) \cdot P_m(x) + \left(x \cdot P_i(x) + 1 \right) \\
     & = x \cdot R_i(x) \cdot P_m(x) + P_{i+1}(x) \label{1992-3:eq:2}
  \end{align}
  である．
  従って，$P_{k+1}(x)$を$P_m(x)$で割ったあまりは，$P_{i+1}(x)$を$P_m(x)$で割ったあまりに等しい．
  $i=0,1,\cdots, m-1$であることに注意する．

  $i=0,1,\cdots,m-2$のとき，$P_{i+1}(x)$を$P_m(x)$で割ったあまりは$P_{i+1}(x)$自身である．

  $i=m-1$の時は$P_{i+1}(x)=P_{m}(x)$であるから，$P_m(x)$で割った余りは$P_0(x)=0$である．

  以上から，いずれの場合も$\diamondsuit$が成り立つ．
  よって$n=k+1$でも$\diamondsuit$は成立する．

  \vspace{5pt}
  以上から，数学的帰納法によって，任意の自然数$n,m$について$\diamondsuit$が成立する．$\cdots$(答)

  \vspace{10pt}
  (2)
  新しく
  \begin{align*}
    Q_n(x) = (x-1)P_n(x)
  \end{align*}
  とおくと，
  \begin{align*}
    Q_n(x)
     & = (x-1)\sum_{k=0}^{n-1}x^k \\
     & = x^n-1
  \end{align*}
  を満たす．
  題意の式
  \begin{align}
    P_l(x) P_m(x^2) P_n(x^4) = P_{100}(x) \label{1992-3:eq:3}
  \end{align}
  について，$P$が多項式だから，$x\neq 1$で成り立てば$x=1$でも成立する．
  そこで$x \neq 1$として，両辺に$(x-1)(x^2-1)(x^4-1)$をかけると
  \begin{align}
     & \text{\cref{1992-3:eq:3}}                                                                                             \\
     & \iff (x-1)P_l(x) \cdot (x^{2}-1)P_m(x^{2}) \cdot (x^{4}-1)P_n(x^{4}) = P_{100}(x)(x-1)(x^2-1)(x^4-1)                  \\
     & \iff Q_l(x) Q_m(x^{2}) Q_n(x^{4}) = Q_{100}(x) \cdot (x^{2}-1) (x^{4}-1)                                              \\
     & \iff (x^l-1) (x^{2m}-1) (x^{4n}-1) = \left(x^{100}-1\right) \left(x^1-1\right) \left(x^2-1\right) \label{1992-3:eq:4}
  \end{align}
  \cref{1992-3:eq:4}が$x$についての恒等式だから次数を比較すると
  \begin{align}
    l+2m+4n           & = 100+2+4 = 106 \label{1992-3:eq:5} \\
    (2m+4n,l+4n,l+2m) & = (104,102,6)   \label{1992-3:eq:6} \\
    (l,2m,4n)         & = (100,4,2) \label{1992-3:eq:7}
  \end{align}
  である．ただし組み合わせについては順不同である．

  \cref{1992-3:eq:7}において，$n$が自然数ゆえ$4n=2$となることはないから，
  \begin{align*}
    4n = 4,  100 \\
    \therefore
    n = 1, 25
  \end{align*}
  である．それぞれの場合について検討する．

  \vspace{5pt}
  \textbf{1$^\circ$ $n=1$の時} \\
  ，$(l,m) = (100,1), (2,50)$ が必要で，共に\cref{1992-3:eq:5,1992-3:eq:6}をみたして十分．

  \vspace{5pt}
  \textbf{2$^\circ$ $n=25$の時} \\
  \cref{1992-3:eq:7}から，$(l,m) = (2,2), (4,1)$ が必要で，共に\cref{1992-3:eq:5,1992-3:eq:6}をみたして十分．

  \vspace{5pt}
  以上から，求める組み合わせは
  \begin{align*}
    (l,m,n) = (100,1,1), (2,50,1), (2,2,25), (4,1,25)
  \end{align*}
  である．$\cdots$(答)
