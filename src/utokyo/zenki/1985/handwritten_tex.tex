\subsection*{第1問}

\noindent
\textbf{[解]} $y = A \sin x$ と $y = 1$ は $0 < x \le \frac{\pi}{2}$ において共有点を持つ。このとき座標として
\[
A \sin t = 1 \quad \left(0 < t \le \frac{\pi}{2}\right) \quad \dots \text{①}
\]

\noindent
又,
\begin{align*}
S_2 - S_1 &= \text{長方形} - 2 S_1 = \int_0^{\frac{\pi}{2}} A \sin x \, dx - 2 \int_t^{\frac{\pi}{2}} (A \sin x - 1) \, dx \\
&= A - 2 A \int_t^{\frac{\pi}{2}} \sin x \, dx + 2 \int_t^{\frac{\pi}{2}} \, dx \\
&= A - 2 A \cos t + \pi - 2 t \equiv f(t)
\end{align*}
とおき, この最大値を求める。①から（以下 $\sin t = s, \cos t = c$）
\begin{align*}
f(t) &= \frac{1}{\sin t} - 2 \frac{\cos t}{\sin t} - 2 t + \pi \\
f'(t) &= - \frac{\cos t}{\sin^2 t} + 2 \frac{1}{\sin^2 t} - 2 = \frac{2 - \cos t}{\sin^2 t} - 2 \quad \left( = \frac{c(2c-1)}{1-c^2} \right) \\
f''(t) &= \frac{s^3 - 2 s c (2-c)}{s^4} = \frac{c^2 - 4c + 1}{s^3}
\end{align*}
下表を考える（$\cos t_0 = 2 - \sqrt{3}$）

\begin{center}
\begin{tabular}{c|cccc}
$t$ & $0$ & & $t_0$ & $\pi/2$ \\ \hline
$c$ & & $-$ & $2-\sqrt{3}$ & $+$ & $0$ \\
$f''$ & & $\searrow$ & $0$ & $\nearrow$ & $1$ \\
$f'$ & $+\infty$ & & $-$ & & $0$
\end{tabular}
\end{center}

\noindent
よって, $f'(t_1) = 0$ なる $t_1$ が $0 < t_1 < t_0$ に存在し, 下表を考える

\begin{center}
\begin{tabular}{c|ccccc}
$t$ & $0$ & & $t_1$ & & $\pi/2$ \\ \hline
$f'$ & & $+$ & $0$ & $-$ & $0$ \\ \hline
$f$ & & $\nearrow$ & & $\searrow$ & $\pi-1$
\end{tabular}
\end{center}

\noindent
よって, $\max f(t) = f(t_1)$ であり, $t_1$ は
\[
f'(t_1) = 0 \text{ より, } \cos t_1 = c' \text{ として } (c' \neq 0)
\]
\[
\frac{2 - c'}{1 - {c'}^2} = 2 \quad \therefore c' = \frac{1}{2}
\]
つまり, 値域から $t_1 = \frac{\pi}{3}$ だから
\[
\max(S_2 - S_1) = f\left(\frac{\pi}{3}\right) = \frac{\pi}{3}
\]
\[
a = \frac{1}{\sin \frac{\pi}{3}} = \frac{2}{3}\sqrt{3}
\]

\subsection*{第3問}

\noindent
\textbf{[解]} 題意から, 2平面は $Z = \frac{1}{t} Y$, $Z = \frac{1}{t} X$ となるから, $Q(t, Y, Z)$ とおける。$RQ$ の傾き及び中点について
\[
\begin{cases}
\cdot M\left(t, \dfrac{Y+t}{2}, \dfrac{Z}{2}\right) \text{ が } Z = \dfrac{1}{t} Y \text{ 上} \\[1ex]
\cdot \vec{RQ} \cdot \begin{pmatrix} 0 \\ 1 \\ -t \end{pmatrix} = 0
\end{cases}
\dots \begin{cases} Y = \dfrac{t(t^2-1)}{1+t^2} \\[1.5ex] Z = \dfrac{2t^2}{1+t^2} \end{cases} \quad \dots \text{①}
\]
となる。対称性から $R(Y, t, Z)$ となる。この時, $QR$ の中点を $N$ とすると, 平面 $OPN$ に関して立体は対称で, $N\left(\frac{t+Y}{2}, \frac{t+Y}{2}, Z\right)$ となるから, もとめる体積 $V$ として
\begin{align*}
V &= 2 \Delta R\text{-}OPN \\
&= 2 \cdot \frac{1}{3} \cdot RN \times \Delta OPN \quad (\because OPN \perp RN) \quad \dots \text{②}
\end{align*}
と書ける。
\[
RN = \sqrt{\left(Y - \frac{t+Y}{2}\right)^2 + \left(t - \frac{t+Y}{2}\right)^2} = \frac{1}{\sqrt{2}} |Y - t| = \frac{1}{\sqrt{2}} \frac{2t}{1+t^2}
\]
\[
\Delta OPN = \frac{1}{2} \cdot \sqrt{2}t \cdot Z = \frac{1}{\sqrt{2}} \frac{2t^3}{1+t^2}
\]
を②に代入して
\[
V = \frac{2}{3} \cdot \frac{1}{\sqrt{2}} \frac{2t}{1+t^2} \cdot \frac{1}{\sqrt{2}} \frac{2t^3}{1+t^2} = \frac{4}{3} \frac{t^4}{(1+t^2)^2}
\]

\textbf{[解2] (後半部)} \\

\begin{align*}
V &= \frac{1}{6} |t Y Z + t Z Y - t Z t - t t Z| \\

&= \frac{4}{3} \frac{t^4}{(1+t^2)^2}
\end{align*}

\subsection*{第5問}

\noindent
\textbf{[解]} $X=m, Y=n$ である事象を $X_m, Y_n$ とおく。$X_m, Y_n$ が独立だから
\[
P(X_m \wedge Y_n) = P(X_m) P(Y_n) \quad \dots \text{①}
\]
又, $X+Y=n$ となる確率が $(n+1) P_{n+1}$ に等しく, この時 $(X, Y) = (k, n-k) \ (k=0, 1, \dots, n)$ だから
\begin{align*}
& P(X+Y=n) = (n+1) P_{n+1} \\
\therefore & \sum_{k=0}^n P(X_k \wedge Y_{n-k}) = (n+1) P_{n+1} \quad \dots \text{②}
\end{align*}
①を②に代入して
\[
\sum_{k=0}^n P_k P_{n-k} = (n+1) P_{n+1} \quad \dots \text{③}
\]
以下, 任意の $n \in \mathbb{Z}_{\ge 0}$ に対し, $P_n = P_0^{n+1} \dots \text{◆}$ であること $\dots$ ④ を帰納的に示す。 \\
$n=0$ の時は明らかに成立する。以下 $n \le l \ (l \in \mathbb{Z}_{\ge 0})$ での◆の成立を仮定する。 \\
③から
\[
(l+1) P_{l+1} = \sum_{k=0}^l P_0^{k+1} P_0^{l-k+1} = (l+1) P_0^{l+2}
\]
$l+1 \neq 0$ から
\[
P_{l+1} = P_0^{l+2}
\]
となり, $n=l+1$ でも◆は成立。以上から④は示された。これを $\sum_{n=0}^\infty P_n = 1$ に代入する。 \\
$P_0 = 1$ は明らかに不適なので, $P_0 \neq 1$ とすると,
\[
\sum_{n=0}^N P_n = P_0 \frac{1 - P_0^{N+1}}{1 - P_0} \quad \dots \text{⑤}
\]
これが収束するのは, $|P_0| < 1$ の時で, このもとで⑤は $\frac{P_0}{1-P_0}$ に収束する。
\[
\frac{P_0}{1-P_0} = 1 \iff P_0 = \frac{1}{2} \quad (|P_0| < 1 \text{をみたす})
\]
だから◆に代入して
\[
P_n = \left(\frac{1}{2}\right)^{n+1}
\]
だから, $r = \frac{1}{2}$ として
\[
\sum_{k=0}^n k \cdot P_k = \frac{1}{1-r} r - \frac{1}{(1-r)^2} r^{n+1} + \frac{r}{1-r} \xrightarrow{n \to \infty} \frac{r}{1-r} = 1 \quad (|r| < 1)
\]
となる。
