\textbf{[解]} 題意から, 2平面は $Z = \frac{1}{t} Y$, $Z = \frac{1}{t} X$ となるから, $Q(t, Y, Z)$ とおける。$RQ$ の傾き及び中点について
\[
\begin{cases}
\cdot M\left(t, \dfrac{Y+t}{2}, \dfrac{Z}{2}\right) \text{ が } Z = \dfrac{1}{t} Y \text{ 上} \\[1ex]
\cdot \vec{RQ} \cdot \begin{pmatrix} 0 \\ 1 \\ -t \end{pmatrix} = 0
\end{cases}
\dots \begin{cases} Y = \dfrac{t(t^2-1)}{1+t^2} \\[1.5ex] Z = \dfrac{2t^2}{1+t^2} \end{cases} \quad \dots \text{①}
\]
となる。対称性から $R(Y, t, Z)$ となる。この時, $QR$ の中点を $N$ とすると, 平面 $OPN$ に関して立体は対称で, $N\left(\frac{t+Y}{2}, \frac{t+Y}{2}, Z\right)$ となるから, もとめる体積 $V$ として
\begin{align*}
V &= 2 \Delta R\text{-}OPN \\
&= 2 \cdot \frac{1}{3} \cdot RN \times \Delta OPN \quad (\because OPN \perp RN) \quad \dots \text{②}
\end{align*}
と書ける。
\[
RN = \sqrt{\left(Y - \frac{t+Y}{2}\right)^2 + \left(t - \frac{t+Y}{2}\right)^2} = \frac{1}{\sqrt{2}} |Y - t| = \frac{1}{\sqrt{2}} \frac{2t}{1+t^2}
\]
\[
\Delta OPN = \frac{1}{2} \cdot \sqrt{2}t \cdot Z = \frac{1}{\sqrt{2}} \frac{2t^3}{1+t^2}
\]
を②に代入して
\[
V = \frac{2}{3} \cdot \frac{1}{\sqrt{2}} \frac{2t}{1+t^2} \cdot \frac{1}{\sqrt{2}} \frac{2t^3}{1+t^2} = \frac{4}{3} \frac{t^4}{(1+t^2)^2}
\]

\textbf{[解2] (後半部)} \\
サラスの公式から
\begin{align*}
V &= \frac{1}{6} |t Y Z + t Z Y - t Z t - t t Z| \\
&= \frac{t}{3} |Y Z - t Z| \\
&= \frac{t}{3} \left| \frac{t(t^2-1)}{1+t^2} - t \right| \frac{2t^2}{1+t^2} \\
&= \frac{t}{3} \cdot \frac{2t}{1+t^2} \cdot \frac{2t^2}{1+t^2} \\
&= \frac{4}{3} \frac{t^4}{(1+t^2)^2}
\end{align*}
