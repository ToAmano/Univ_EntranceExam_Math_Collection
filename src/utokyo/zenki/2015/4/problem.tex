数列$\{ p_n \}$を次のように定める。
\begin{align*}
  p_1=1, \hspace{1\zw} p_2=2, \hspace{1\zw}
p_{n+2}=\frac{{p_{n+1}}^2+1}{p_n} \hspace{1\zw} (n=1, \, 2, \, 3, \, \cdots)
\end{align*}
\begin{enumerate}
  \item $\displaystyle\frac{{p_{n+1}}^2+{p_n}^2+1}{p_{n+1}p_n}$が$n$によらないことを示せ。
  \item すべての$n=2, \, 3, \, 4, \, \cdots$に対し，$p_{n+1}+p_{n-1}$を$p_n$のみを使って表せ。
  \item 数列$\{ q_n \}$を次のように定める。
\begin{align*}
  q_1=1, \hspace{1\zw} q_2=1, \hspace{1\zw}
q_{n+2}=q_{n+1}+q_n \hspace{1\zw} (n=1, \, 2, \, 3, \, \cdots)
\end{align*}
すべての$n=1, \, 2, \, 3, \, \cdots$に対し，$p_n=q_{2n-1}$を示せ。
\end{enumerate}
