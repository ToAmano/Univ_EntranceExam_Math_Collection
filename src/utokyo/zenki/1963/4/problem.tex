一辺の長さ $a$ の正四面体 $ABCD$ の辺 $AB$，$AC$，$AD$ の上に
$A$ から等距離にそれぞれ点 $P$，$Q$，$R$ をとり，
$P$，$Q$，$R$ から面 $BCD$ に下した垂線の足をそれぞれ $P'$，$Q'$，$R'$ とする．

\begin{figure}[htbp]
  \centering
  \begin{tikzpicture}[scale=1.5, >=stealth]
    \coordinate (A) at (0, 2.5);

    \coordinate (B) at (-1.8, 0);
    \coordinate (C) at (-0.4, -0.8);
    \coordinate (D) at (1.8, 0);

    \draw[dashed] (B) -- (D);

    \draw[thick] (A) -- (B) -- (C) -- (D) -- (A);
    \draw[thick] (A) -- (C);

    \coordinate (P) at ($(A)!0.4!(B)$);
    \coordinate (Q) at ($(A)!0.4!(C)$);
    \coordinate (R) at ($(A)!0.4!(D)$);

    \draw[thick] (P) -- (Q) -- (R) -- (P);

    \draw[dashed] (P) -- (-0.8, -0.4) node[below] {$P'$};
    \draw[thick] (Q) -- (-0.2, -0.7) node[below] {$Q'$};
    \draw[thick] (R) -- (0.8, -0.4) node[below] {$R'$};

    \node[above] at (A) {$A$};
    \node[left] at (B) {$B$};
    \node[below left] at (C) {$C$};
    \node[right] at (D) {$D$};

    \node[left] at (P) {$P$};
    \node[left] at (Q) {$Q$};
    \node[right] at (R) {$R$};
  \end{tikzpicture}
\end{figure}

\begin{enumerate}
  \item 三角柱 $PQR-P'Q'R'$ の体積が最大になるときの $AP$ の長さを求めよ．
  \item この三角柱の体積の最大値 $V_0$ と正四面体 $ABCD$ の体積 $V$ の比 $\displaystyle\frac{V_0}{V}$ を求めよ．
\end{enumerate}
