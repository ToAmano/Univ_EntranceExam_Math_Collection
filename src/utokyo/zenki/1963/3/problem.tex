$AD=a$，$BC=b$，$AD//BC$ なる台形がある．
対角線 $AC$，$BD$ の交点を $P_1$ とし，$CD$ 上に点 $Q_1$ を $P_1Q_1//AD$ となるようにとる．
次に，$AQ_1$，$DP_1$ の交点を $P_2$ とし，$CD$ 上に点 $Q_2$ を $P_2Q_2//AD$ となるようにとる．
次に，$AQ_2$，$DP_2$ の交点を $P_3$ とし，$CD$ 上に点 $Q_3$ を $P_3Q_3//AD$ となるようにとる．
以下同様にくりかえして，$n$ 回目にできる線分 $P_nQ_n$ の長さを $x_n$ とするとき

\begin{figure}[htbp]
  \centering
  \begin{tikzpicture}[scale=1.5, >=stealth]
    \coordinate (A) at (0.4, 2.5);
    \coordinate (D) at (2.4, 2.5);
    \coordinate (B) at (0, 0);
    \coordinate (C) at (3.2, 0);

    \draw[thick] (A) -- (D) -- (C) -- (B) -- cycle;
    \draw[thick] (A) -- (C);

    \coordinate (Q1) at (intersection of P1--+(1,0) and C--D);

    \draw[thick] (A) -- (Q1);
    \coordinate (P2) at (intersection of A--Q1 and B--D);
    \coordinate (Q2) at (intersection of P2--+(1,0) and C--D);
    \draw[thick] (P2) -- (Q2);

    \draw[thick] (A) -- (Q2);
    \coordinate (P3) at (intersection of A--Q2 and B--D);
    \coordinate (Q3) at (intersection of P3--+(1,0) and C--D);
    \draw[thick] (P3) -- (Q3);

    \node[above left] at (A) {$A$};
    \node[above right] at (D) {$D$};

    \node[below right] at (C) {$C$};

    \node[below left] at (P1) {$P_1$};
    \node[below right] at (P2) {$P_2$};
    \node[above left] at (P3) {$P_3$};

    \node[right] at (Q1) {$Q_1$};
    \node[right] at (Q2) {$Q_2$};
    \node[right] at (Q3) {$Q_3$};
  \end{tikzpicture}

\begin{enumerate}
  \item $x_n$ を $a$，$b$，$n$ で表わせ．
  \item $\triangle DP_{n+1}Q_n$ の面積を $F_n$ とするとき
  $\displaystyle\sum_{n=1}^\infty F_n$ を求めよ．
\end{enumerate}
