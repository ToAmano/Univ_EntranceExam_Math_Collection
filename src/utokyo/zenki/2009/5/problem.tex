\begin{enumerate}
  \item 実数$x$が$-1<x<1$，$x\neq0$をみたすとき，次の不等式を示せ．
\begin{align*}
  {(1-x)}^{1-\frac{1}{x}}<{(1+x)}^\frac{1}{x}
\end{align*}
  \item 次の不等式を示せ．
\begin{align*}
  {0.9999}^{101}<0.99<{0.9999}^{100}
\end{align*}
\end{enumerate}
