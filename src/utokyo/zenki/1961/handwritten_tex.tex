\section*{第1問}

\begin{proof}[解]
$xy$平面で $O(0,0)$, $A(1,0)$, $B\left(\dfrac{1}{2}, \dfrac{\sqrt{3}}{2}\right)$ とする．
時刻 $t$ での $P, Q, R$ の座標は，
\begin{align*}
  &P(t+1, 0) \\
  &Q\left(\frac{3}{2}t, \frac{\sqrt{3}}{2}t\right) \\
  &R\left(t+\frac{1}{2}, \sqrt{3}t+\frac{\sqrt{3}}{2}\right)
\end{align*}
で与えられる．

\begin{center}
\begin{tikzpicture}[scale=1.2, >=stealth]
  \draw[->] (-0.3,0) -- (2.2,0) node[right] {$x$};
  \draw[->] (0,-0.3) -- (0,1.8) node[above] {$y$};
  \node[below left] at (0,0) {$O$};
  \draw[line width=0.6pt] (0,0) -- (2.0,0) node[below] {$X$};
  \draw[line width=0.6pt] (0,0) -- (60:2.0) node[above right] {$Y$};
  \draw[line width=0.6pt] (0,0) -- (30:2.2) node[right] {$Z$};
  \draw (1,0.05) -- (1,-0.05) node[below] {$1$};
\end{tikzpicture}
\end{center}

\begin{enumerate}[(1)]
  \item $\vec{PQ} = \begin{pmatrix} \frac{1}{2}t-1 \\ \frac{\sqrt{3}}{2}t \end{pmatrix}$, $\vec{PR} = \begin{pmatrix} -\frac{1}{2} \\ \sqrt{3}t+\frac{\sqrt{3}}{2} \end{pmatrix}$ だから，$P, Q, R$ が一直線上にある時 $k \in \mathbb{R}$ があって
  \[
    \begin{pmatrix} \frac{1}{2}t-1 \\ \frac{\sqrt{3}}{2}t \end{pmatrix} = k \begin{pmatrix} -\frac{1}{2} \\ \sqrt{3}t+\frac{\sqrt{3}}{2} \end{pmatrix}
  \]
  と書ける．
  \[
    \begin{cases}
      \frac{1}{2}t-1 = -\frac{1}{2}k \\
      \frac{\sqrt{3}}{2}t = k \left(\sqrt{3}t + \frac{\sqrt{3}}{2}\right)
    \end{cases}
    \quad \therefore 
    \begin{cases}
      k = \frac{3-\sqrt{5}}{2} \\
      t = \frac{1+\sqrt{5}}{2}
    \end{cases}
  \]
  だから，求めるのは $t = \dfrac{1+\sqrt{5}}{2}$ である．

  \item $\triangle PQR$ の面積は (1) から
  \begin{align*}
    &\frac{1}{2} \left| \left(\frac{1}{2}t-1\right)\sqrt{3}\left(t+\frac{1}{2}\right) + \frac{1}{2}\frac{\sqrt{3}}{2}t \right| \\
    &= \frac{1}{2} \left| \sqrt{3}\left(\frac{1}{2}t^2 - \frac{3}{4}t - \frac{1}{2}\right) + \frac{\sqrt{3}}{4}t \right| \\
    &= \frac{\sqrt{3}}{4} |t^2 - t - 1|
  \end{align*}
  一方，$\triangle OAB$ の面積は $\dfrac{1}{2} \cdot 1 \cdot \dfrac{\sqrt{3}}{2} = \dfrac{\sqrt{3}}{4}$ だから，これらが等しいとき，
  \[
    |t^2 - t - 1| = 1 \quad \therefore t = 1, 2
  \]
\end{enumerate}
\end{proof}

\section*{第2問}

\begin{proof}[解]
$f\left(-\dfrac{1}{5}\right) = A$, $f\left(-\dfrac{1}{10}\right) = B$, $f(0) = c$, $f\left(\dfrac{1}{10}\right) = D$, $f\left(\dfrac{1}{5}\right) = E$ とおき，
\[
  f(x) = ax^4 + bx^3 + cx^2 + dx + e
\]
とする．

\begin{itemize}
  \item $f(0) = c = e$
  \begin{align}
    \begin{cases}
      f\left(-\frac{1}{5}\right) = A = \frac{a}{625} - \frac{b}{125} + \frac{c}{25} - \frac{d}{5} + e \\[5pt]
      f\left(\frac{1}{5}\right) = E = \frac{a}{625} + \frac{b}{125} + \frac{c}{25} + \frac{d}{5} + e
    \end{cases} \label{eq:1961_2_1} \\[5pt]
    \begin{cases}
      f\left(\frac{1}{10}\right) = D = \frac{a}{10^4} + \frac{b}{10^3} + \frac{c}{10^2} + \frac{d}{10} + e \\[5pt]
      f\left(-\frac{1}{10}\right) = B = \frac{a}{10^4} - \frac{b}{10^3} + \frac{c}{10^2} - \frac{d}{10} + e
    \end{cases} \label{eq:1961_2_2}
  \end{align}
\end{itemize}

\eqref{eq:1961_2_1}, \eqref{eq:1961_2_2} の辺々引いて
\[
  \begin{cases}
    E - A = \frac{2b}{125} + \frac{2}{5}d \\[5pt]
    D - B = \frac{2b}{1000} + \frac{2}{10}d
  \end{cases}
\]
$b$ を消去して
\begin{align*}
  \frac{6}{5}d &= -A - 8B + 8D - E \\
  &= 3.258
\end{align*}
\[
  \therefore d = 2.715
\]
より，
\[
  f'(0) = 2.715
\]
\end{proof}

\section*{第3問}

\begin{proof}[解]
直円錐の軸を含む平面で切る．右図のように点，図形量をおく．($0 < \theta < \frac{\pi}{2}$) 以下 $s = \sin\theta$, $c = \cos\theta$ と書く．

\begin{center}
\begin{tikzpicture}[scale=1.5, >=stealth]
  \coordinate (O) at (0,0);
  \coordinate (H) at (0,2);
  \coordinate (A) at (-1.5,0);
  \coordinate (A2) at (1.5,0);
  
  \draw[thick] (A) -- (A2) -- (H) -- cycle;
  \draw[dashed] (O) -- (H);
  
  \node[circle, draw, minimum size=1.3cm] (C) at (0,0.65) {};
  
  \draw (0,0.15) -- (-0.15,0.15) -- (-0.15,0);
  
  \draw (-1.2,0) arc (0:53:0.3);
  \node at (-1.05,0.15) {$\theta$};
  
  \draw[dashed] (0,0.65) -- (-0.39, 1.17) node[midway, right] {$a$};
  
  \node[below left] at (A) {$A$};
  \node[below] at (O) {$O$};
  \node[above] at (H) {$H$};
\end{tikzpicture}
\end{center}

\[
  AO = \frac{a}{s}, \quad AH = \frac{a}{cs}, \quad OH = \frac{a}{c}
\]
だから，直円錐の表面積 $T$ は
\begin{align*}
  T &= \pi \overline{AO}^2 + \pi \overline{AO} \cdot \overline{AH} \\
  &= \pi \left( \frac{a^2}{s^2} + \frac{a^2}{cs^2} \right) \\
  &= \pi a^2 \frac{1}{c(1-c)} \ge \pi a^2 \frac{1}{\frac{1}{2} \cdot \frac{1}{2}} \quad (\text{等号成立は } c = \frac{1}{2} \text{ の時})
\end{align*}
だから，この時
\[
  \overline{OH} = 2a
\]
である．

\begin{framed}
\noindent $\triangleright$ 直円錐の側面積
\begin{minipage}{0.5\textwidth}
\[
  \begin{cases}
    V = \frac{1}{3}\pi r^2 h \\
    S = \pi r \sqrt{r^2 + h^2} = \pi r L
  \end{cases}
\]
\end{minipage}%

\centering

  \draw[dashed] (0,0) -- (0,1.8) node[midway, right] {$h$};
  \draw[dashed] (0,0) -- (1,0) node[midway, below] {$r$};
\end{tikzpicture}
\end{minipage}
\end{framed}
\end{proof}
