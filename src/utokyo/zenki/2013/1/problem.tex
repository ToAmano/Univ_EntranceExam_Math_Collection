\begin{align*}

\end{align*}

  (x_{n+1}, \, y_{n+1})=(ax_n-by_n, \, bx_n+ay_n) \hspace{1\zw} (n=0, \, 1, \, 2, \, \cdots)
\end{align*}

  \item $P_0=P_6$
