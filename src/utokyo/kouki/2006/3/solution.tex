  {\bf[解]}
  (1)
  問題文では$p\in\mathbb{N}$として与えられているが，ここでは$p \in\mathbb{Z} \ge 0$に拡張して考える．
  この時
  \begin{align*}
    S_0
     & = \sum_{k=1}^{n}k^{0} \\
     & = \sum_{k=1}^{n}1     \\
     & = n
  \end{align*}
  と定義すれば$S_0$も定義できる．

  さて，題意を$p$についての帰納法で示す．
  まず $p=0$ の時$S_{0}(n)=n$ とすれば題意は成立する．

  そこで以下 $\ell \in \mathbb{Z}_{\ge 0}$ とし，$0 \le p \le \ell$ の時に題意が成立すると仮定する．
  $\sum_{k=1}^{n} (k+1)^{l+2} - k^{l*2}$ を $2$ 通りで表すと
  \begin{align*}
    (n+1)^{\ell+2} - 1
     & = \sum_{k=1}^{n} \left[ (k+1)^{\ell+2} - k^{\ell+2} \right]                                                         \\
     & = \sum_{t=0}^{\ell+1} {}_{\ell+2}C_{t}\left(\sum_{k=1}^{n}k^t\right)                                                \\
     & = \left(\ell + 2 \right)\sum_{k=1}^{n}k^{\ell+1} + \sum_{t=0}^{\ell} {}_{\ell+2}C_{t}\left(\sum_{k=1}^{n}k^t\right) \\
     & = \left(\ell + 2 \right)\sum_{k=1}^{n}k^{\ell+1} + \sum_{t=0}^{\ell} {}_{\ell+2}C_{t} S_{t}(n)
  \end{align*}
  だから，整理して
  \begin{align}
    \sum_{k=1}^{n}k^{\ell+1} = \frac{1}{\ell+2} \left( (n+1)^{\ell+2} - 1 - \sum_{t=0}^{\ell} {}_{\ell+2}C_{t} S_{t}(n)\right) \label{2006-3:eq:1}
  \end{align}
  を得る．

  仮定より，$S_{t}(n)\, (t=0,1,\cdots,\ell)$ は $t$ 次多項式なので$\sum_{k=1}^{n}k^{\ell+1}$の最高次は第一項目である．
  よって$\sum_{k=1}^{n}k^{\ell+1}$は$\ell+2$次多項式であり，題意は $P=\ell+1$ でも成立する．

  以上より数学的帰納法によって題意は示された．$\cdots$(答)

  \vspace{10pt}
  (2)
  $S_1(0)=0$ 及び\cref{2006-3:eq:1}から帰納的に $S_{p}(0)=0$ であるから，
  題意の条件式
  \begin{align}
    \sum_{j=1}^{q} a_{j} S_{2j-1}(x) = x^{q}(x+1)^{q} \label{2006-3:eq:2}
  \end{align}
  は$a_{j}$に限らず $x=0$ で成立する．

  したがって，
  \begin{center}
    \text{"(1)が $x$ に関する恒等式"}
  \end{center}
  $\Leftrightarrow$ \text{"階差をとった}
  \begin{align}
    \sum_{j=1}^{q} a_{j} \{ S_{2j-1}(x) - S_{2j-1}(x-1) \} = x^{q}(x+1)^{q} - (x-1)^{q}x^{q} \label{2006-3:eq:3}
  \end{align}
  が $x$ に関する恒等式"

  である．
  なぜならば$x=1$とすると$x=0$での成立から階差を取る前の式が成立し，
  以下帰納的に$x=2,3,\cdots$で成立する．
  $S_{p}(n)$が多項式だから与えられた式が恒等式になるからである．

  この両辺は $x$ の多項式だから，\cref{2006-3:eq:3} が $2$ 以上の自然数 $n$ のときに成立することが必要十分である．

  \cref{2006-3:eq:3}の左辺に関しては
  \begin{align}
    S_{2j-1}(n) - S_{2j-1}(n-1)
     & = \sum_{k=1}^{n}k^{2j-1} - \sum_{k=1}^{n-1}k^{2j-1} \\
     & = n^{2j-1} \label{2006-3:eq:4}
  \end{align}
  と表せる．

  一方\cref{2006-3:eq:3}の右辺に関しては二項展開すると
  \begin{align}
    (x+1)^{q} - (x-1)^{q} = \sum_{k=0}^{q} \{ 1 - (-1)^{q-k} \} {}_{q}C_{k} x^{k} \label{2006-3:eq:5}
  \end{align}
  である．

  以上\cref{2006-3:eq:4,2006-3:eq:5}を\cref{2006-3:eq:3}に代入して
  \begin{align*}
    \sum_{j=1}^{g} a_{j} x^{2j-1} = \sum_{k=0}^{q} \{ 1 - (-1)^{q-k} \} {}_{q}C_{k} x^{k+q}
  \end{align*}
  となる．従って，$2j-1=k+q$次の係数を比較する．この時$k=2j-q-1$だから
  \begin{align*}
    a_{j}
     & = \{ 1 - (-1)^{q-k} \} {}_{q}C_{k}             \\
     & = \{ 1 - (-1)^{q-(2j-q-1)} \} {}_{q}C_{2j-q-1} \\
     & = \{ 1 - (-1)^{-2j+1)} \} {}_{q}C_{2j-q-1}     \\
     & = 2{}_{q}C_{2j-q-1}
  \end{align*}
  となる．$\cdots$(答)

  (3)
  (2) と同じく，
  \begin{align}
    \sum_{j=1}^{q-1} b_{j} S_{2j}(x) = x^{q-1}(x+1)^{q-1}(cx+q) \label{2006-3:eq:6}
  \end{align}
  は $x=0$ で成立するから，\cref{2006-3:eq:6}の階差をとった
  \begin{align*}
    \sum_{j=1}^{q-1} b_{j} \{ S_{2j}(n) - S_{2j}(n-1) \}
     & = n^{q-1}(n+1)^{q-1}(cn+q)           \\
     & \quad - (n-1)^{q-1}n^{q-1}(c(n-1)+q)
  \end{align*}
  が 2 以上の自然数 $n$ で成立すれば良い．
  左辺は$S_{2j}(n) - S_{2j}(n-1) = n^{2j}$を用いて簡単にできて，
  \begin{align}
    \sum_{j=1}^{q-1} b_{j} n^{2j}
     & = n^{q-1} \left[ cn\left\{(n+1)^{q-1} - (n-1)^{q-1} \right\}\right.       \\
     & \quad + \left.q(n+1)^{q-1} + (c-q) (n-1)^{q-1} \right]\label{2006-3:eq:7}
  \end{align}
  となる．
  左辺が偶数冪のみからなるから，右辺もそうならないといけない．
  左辺第一項は偶数次だから，残りの第二・第三項が偶数冪である．
  従って
  \begin{align*}
    n^{q-1}\left[q(n+1)^{q-1} + (c-q) (n-1)^{q-1}\right]
  \end{align*}
  が偶数冪である．二項定理で展開すると
  \begin{align*}
    \sum_{k=0}^{q-1}{}_{q-1}C_k\left[q + (c-q)(-1)^{q-1-k}\right]n^{q+k-1}
  \end{align*}
  で，これが偶数冪のみからなる条件は
  \begin{align}
     & q - (c-q) = 0              \\
    \therefore
     & c = 2q \label{2006-3:eq:8}
  \end{align}
  である．
  これを\cref{2006-3:eq:7}に代入して
  \begin{align}
    \sum_{j=1}^{q-1} b_{j} n^{2j}
     & = qn^{q-1} \left[ 2n\left\{(n+1)^{q-1} - (n-1)^{q-1} \right\}\right.    \nonumber  \\
     & \quad + \left.\left\{(n+1)^{q-1} + (n-1)^{q-1}\right\} \right] \label{2006-3:eq:9}
  \end{align}
  を得る．右辺の各項は二項定理で展開して
  \begin{align*}
    (n+1)^{q-1} - (n-1)^{q-1}
     & = \sum_{k=0}^{q-1} {}_{q-1}C_{k} \left[1-(-1)^{q-1-k}\right]n^{k}   \\
    (n+1)^{q-1}+(n-1)^{q-1}
     & = \sum_{k=0}^{q-1} {}_{q-1}C_{k} \left[1 + (-1)^{q-1-k}\right]n^{k}
  \end{align*}
  だから\cref{2006-3:eq:9}に代入すると
  \begin{align*}
    \sum_{j=1}^{q-1} b_{j} n^{2j}
     & = q \sum_{k=0}^{q-1} {}_{q-1}C_{k} \left[2\left[1-(-1)^{q-1-k}\right] n^{k+q} \right. \\
     & \quad + \left.\left[1 + (-1)^{q-1-k}\right]n^{k+q-1}\right]
  \end{align*}
  となる．
  両辺で$n^{2j}$の項を係数比較すると，
  \begin{align*}
    b_{j}
     & = 4q{}_{q-1}C_{2j-q} + 2q{}_{q-1}C_{2j-q+1}
  \end{align*}
  である．ここで二項係数の性質から
  \begin{align*}
     & q {}_{q-1}C_{2j-q} = (2j-q+1){}_{q}C_{2j-q+1}   \\
     & q{}_{q-1}C_{2j-q+1} = (2q-2j-1){}_{q}C_{2j-q+1}
  \end{align*}
  だから代入して
  \begin{align*}
    b_{j}
     & = 4(2j-q+1){}_{q}C_{2j-q+1} + 2q{}_{q-1}C_{2j-q+1}        \\
     & = 2(4j-2q+2){}_{q}C_{2j-q+1} + 2(2q-2j-1){}_{q}C_{2j-q+1} \\
     & = 2(2j+1){}_{q}C_{2j-q+1}
  \end{align*}
  である．
  以上から
  \begin{align*}
    \begin{dcases}
      c = 2q \\
      b_{j} = 2(2j+1){}_{q}C_{2j+1-q}
    \end{dcases}
  \end{align*}
  である．
  $\cdots$(答)

  (4)
  $r \in \mathbb{Z}_{\ge 2}$とする．
  $P=2r-1$ に対して，
  \begin{align}
    S'_{2r-1}(x) = (2r-1)S_{2r-2}(x) \label{2006-3:eq:10}
  \end{align}
  が成立することを $r$ についての帰納法で示す．

  $r=2$ の時は，
  \begin{align*}
    S_3'(x)
     & = \left[\frac{1}{4}x^2(x+1)^2\right]' \\
     & = \frac{1}{2}x(x+1)(2x+1)             \\
     & = 3S_2(x)
  \end{align*}
  で成立する．

  そこで以下 $q \ge 3$ として，$2 \le r \le q-1$ で\cref{2006-3:eq:10}の成立を仮定する．
  (2) の両辺を $x$ で微分して，
  \begin{align*}
    \frac{d}{dx} \sum_{j=1}^{q} a_j S_{2j-1}(x)
     & = \left[x^{q}(x+1)^{q}\right]'          \\
     & = qx^{q-1}(x+1)^{q} + qx^{q}(x+1)^{q-1} \\
     & = q(2x+1)x^{q-1}(x+1)^{q-1}
  \end{align*}
  となるが(3)の結果より，これは$\sum_{j=1}^{q-1} b_{j} S_{2j}(x)$に等しい，すなわち
  \begin{align*}
    \sum_{j=1}^{q} a_j S'_{2j-1}(x) = \sum_{j=1}^{q-1} b_{j} S_{2j}(x)
  \end{align*}
  である．
  さらに整理して
  \begin{align*}
    \sum_{j=1}^{q-1} a_j S'_{2j-1}(x) + a_{q}S'_{2q-1}(x) = \sum_{j=1}^{q-2} b_{j} S_{2j}(x) + b_{q-1} S_{2q-2}(x)
  \end{align*}
  $a_1=0$だから左辺の和は$j=1$からに変更できて
  \begin{align*}
    \sum_{j=2}^{q-1} a_j S'_{2j-1}(x) + a_{q}S'_{2q-1}(x) = \sum_{j=1}^{q-2} b_{j} S_{2j}(x) + b_{q-1} S_{2q-2}(x)
  \end{align*}
  ここで，(2)(3)の結果より$b_{j-1} = (2j-1)a_{j}$だから，仮定と合わせて
  \begin{align*}
    \sum_{j=2}^{q-1} a_j S'_{2j-1}(x) = \sum_{j=1}^{q-2} b_{j} S_{2j}(x)
  \end{align*}
  と和の部分は左右でキャンセルするので
  \begin{align*}
     & a_{q}S'_{2q-1}(x) =  b_{q-1} S_{2q-2}(x)     \\
     & a_{q}S'_{2q-1}(x) =  (2q-1)a_{q} S_{2q-2}(x)
  \end{align*}
  となる．$a_{q}\neq 0$だから
  \begin{align*}
    S'_{2q-1}(x) =  (2q-1) S_{2q-2}(x)
  \end{align*}
  となり，$r=q$でも題意は成立する．

  以上から数学的帰納法により，任意の$3$以上の奇数について題意が成立することが示された．  $\cdots$(答)
