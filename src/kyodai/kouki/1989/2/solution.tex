  {\bf[解]}

  立体を $xy$平面で切断すると ($r>1/2$に注意) と
  対称性より図は\cref{1989-2:fig:1}のようになる．$\star$

  \begin{figure}[H]
    \centering
    \begin{tikzpicture}
      \begin{axis}[
          axis lines=middle,
          unit vector ratio*=1 1, % xとyの単位長さを同じにする
          xmin=-1.5, xmax=1.5,
          ymin=-0.2, ymax=3, % 図形が収まるようにy軸の範囲を調整
          xlabel=$x$, ylabel=$y$,
          xtick={-0.866,0,0.866}, % tの値をx軸にマーク
          xticklabels={$-t$,$O$,$t$},
          ytick={0.75}, % 1/4, 3/4, 5/4をy軸にマーク
          yticklabels={$t^2$},

        ]

        % パラメータをr=1に基づいて計算
        \pgfmathsetmacro{\r}{1}
        \pgfmathsetmacro{\tval}{sqrt(\r*\r - 1/4)} % t = sqrt(r^2 - 1/4)
        \pgfmathsetmacro{\Yval}{\r*\r + 1/4}     % Y = r^2 + 1/4
        \pgfmathsetmacro{\tsquared}{\tval*\tval} % PとQのy座標 t^2

        % 円の中心 O' (0, Y)

        \addplot[domain=-1.5:1.5, samples=100, blue] {x*x};
        \node[above right, blue] at (1.0, 0.8^2) {$y=x^2$};


        % --- 円 x^2 + (y-Y)^2 = r^2 を描画 ---
        % 中心 (0, Yval)、半径 r の円
        % パラメータ表示: (r*cos(theta), r*sin(theta) + Yval)
        \addplot[domain=0:360, samples=200, red] ({ \r * cos(x) }, { \r * sin(x) + \Yval });


        % % --- 点とラベルを描画 ---
        % \fill (0,0) circle (1.5pt) node[below left] {$O$};
        \fill (Oprime) circle (1.5pt) node[above right] {$O'$}; % 円の中心

        \fill (P) circle (1.5pt) node[above right] {$P$};
        \fill (Q) circle (1.5pt) node[above left] {$Q$};
        \fill (0,0.25) circle (1.5pt);

        % Qからx軸への破線
        \draw[dashed] (Q) -- (\tval, 0);
        \draw[dashed] (P) -- (-\tval, 0);

        \draw[dashed] (Oprime) -- (Q) node[midway, sloped, above] {$r$};
        \draw[dashed] (P) -- (Q);

        % % 塗りつぶす領域は、放物線 y=x^2 と円の下側の弧 y = Y - sqrt(r^2 - x^2) に囲まれた部分。
        % % 境界は x=-t から x=t。
        \addplot[domain=-\tval:\tval, draw=none, name path=A] {x^2};
        \addplot[domain=\tval:-\tval, draw=none, name path=B] (x, {\Yval - sqrt(\r*\r - x*x)});

      \end{axis}
    \end{tikzpicture}
    \caption{$xy$平面での図形の切断面．題意の立体の領域を斜線で塗りつぶしている．}
    \label{1989-2:fig:1}
  \end{figure}

  放物線と円の二接点を，正のパラメータ$t>0$を用いてP$(-t,t^2)$,Q$(t,t^2)$とおく．
  球の中心を$O'(0,Y)$ とおくと点Qが接点である条件は，
  $\vec{l}=(1,2t)$を点Qでの $y=x^2$の接線ベクトルとして
  \begin{align}
     & \begin{dcases}
         \vec{O'Q}^2=r^2 \\
         \vec{O'Q} \perp \vec{l}
       \end{dcases}                                                                                              \\
    \therefore
     & \begin{dcases}
         t^2 + (t^2-Y)^2 = r^2 & \quad \dots \textcircled{1} \\
         \mqty(t                                             \\ t^2-Y) \cdot \mqty(1 \\ 2t) = 0 & \quad \dots \textcircled{2}
       \end{dcases} \\
    \therefore
     & \begin{dcases}
         t^2 + (t^2-Y)^2 = r^2                   \\
         1 + 2(t^2-Y) = 0 & \quad (\because t>0)
       \end{dcases}                                                                             \\
    \therefore
     & \begin{dcases}
         t^2 + \frac{1}{4} = r^2 \\
         t^2-Y = -\frac{1}{2}
       \end{dcases}  \label{1989-2:eq:1}                                                                                   \\
    \therefore
     & \begin{dcases}
         t = \sqrt{r^2-\frac{1}{4}} & \quad (\because t>0) \\
         Y = r^2+\frac{1}{4}
       \end{dcases}
  \end{align}
  となる．

  以下，回転体の体積を求める．
  求める体積は図の斜線部の領域を$y$軸周りに回転させた立体の体積 $V$ に等しい．
  これは二つの部分に分割して考えることができて
  \begin{align}
    V = & (\text{放物線} POQ \text{を} y \text{軸回りに回転した体積 } V_1)                           \\
        & - (\text{円弧} PQ \text{を} y \text{軸回りに回転した体積 } \quad V_2) \label{1989-2:eq:2}
  \end{align}
  となる．以下$V_1,V_2$を別個に求める．

  まず$V_1$にはバームクーヘン積分法を用いると
  \begin{align*}
    V_1
     & = \int_0^t 2\pi x \cdot x^2 dx       \\
     & = \frac{1}{2}\pi t^4                 \\
     & = \frac{1}{2}\pi (r^2-\frac{1}{4})^2
  \end{align*}
  である．

  次に$V_2$は\cref{1989-2:eq:1}から，図のように円$x^2+y^2=r^2$の$1/2\le x\le r$の部分を回転させた体積に等しい．

  \begin{figure}[H]
    \centering
    \begin{tikzpicture}
      \begin{axis}[

          unit vector ratio*=1 1,
          xmin=-1.2, xmax=1.2,

          xtick={-1, 0, 0.5, 1},

          ytick={-1, 0, 1},

          enlargelimits=true,

        % Define r and xmin using \pgfmathsetmacro
        \pgfmathsetmacro{\r}{1}
        \pgfmathsetmacro{\xmin}{0.5}

        \pgfmathsetmacro{\yvalue}{sqrt(\r*\r - \xmin*\xmin)} % 

        % Draw the circle
        \addplot [domain=0:360, samples=200] ({\r*cos(x)}, {\r*sin(x)});
        \addplot [draw=none, name path=B] coordinates {(0.5,-0.866) (0.5,0.866)};
        \addplot [domain=-60:60, draw=none, name path=A] ({\r*cos(x)}, {\r*sin(x)});
        \addplot [pattern=north east lines] fill between [of=A and B];


        % Mark x = 1/2
        \draw [dashed] (\xmin, -\yvalue) -- (\xmin, \yvalue);
        % \node [below] at (\xmin, 0) {1/2};
        % \node [right] at (\r, 0) {r};
        % \node [above] at (0, \r) {r};
        \node [above right] at (0.5, 1) {$x^2+y^2=r^2$};
      \end{axis}
    \end{tikzpicture}
    \caption{求める回転体の部分}
    \label{1989-1:fig:2}
  \end{figure}

  従って
  \begin{align*}
    V_2
     & = \int_{\frac{1}{2}}^r \pi (r^2-x^2) dx                             \\
     & = \pi \left[ r^2x - \frac{1}{3}x^3 \right]_{\frac{1}{2}}^r          \\
     & = \pi \left( \frac{2}{3}r^3 - \frac{1}{2}r^2 + \frac{1}{24} \right)
  \end{align*}
  となる．

  以上$V_1$と$V_2$の表現を\cref{1989-2:eq:2}に代入して
  \begin{align*}
    V
     & = \pi \left[ \frac{1}{2}\left(r^2-\frac{1}{4}\right)^2 - \left(\frac{2}{3}r^3 - \frac{1}{2}r^2 + \frac{1}{24}\right) \right] \\
     & = \pi \left[ \frac{1}{2}r^4 - \frac{1}{4}r^2 + \frac{1}{32} - \frac{2}{3}r^3 + \frac{1}{2}r^2 - \frac{1}{24} \right]         \\
     & = \pi \left[ \frac{1}{2}r^4 - \frac{2}{3}r^3 + \frac{1}{4}r^2 - \frac{1}{96} \right]
  \end{align*}
  が求める面積である．$\cdots$(答)
