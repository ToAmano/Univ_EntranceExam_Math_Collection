$A=\begin{pmatrix} \displaystyle\cos\frac{2\pi}{3} & \displaystyle-\sin\frac{2\pi}{3}
\\ \displaystyle\sin\frac{2\pi}{3} & \displaystyle\cos\frac{2\pi}{3} \end{pmatrix}$とし，
ベクトル$\overrightarrow{u_n}=
\begin{pmatrix} x_n \\ y_n \end{pmatrix}$ $(n=1, \, 2, \, \cdots\cdots)$を次のように定義する．
\begin{align*}
  \overrightarrow{u_1}=\begin{pmatrix} 0 \\ 0 \end{pmatrix}, \hspace{1\zw}
\overrightarrow{u_2}=\begin{pmatrix} 1 \\ 0 \end{pmatrix}, \hspace{1\zw}
\overrightarrow{u_n}=\overrightarrow{u_{n-1}}+
\frac{1}{2}A(\overrightarrow{u_{n-1}}-\overrightarrow{u_{n-2}}) \hspace{1\zw} (n\geqq3)
\end{align*}
\begin{enumerate}
  \item $x_n$，$y_n$を求めよ．
  \item $\displaystyle\lim_{n\to\infty}x_n$，$\displaystyle\lim_{n\to\infty}y_n$を求めよ．
\end{enumerate}
