\section*{第1問}

\begin{proof}[解]
$2^{n-1}\alpha=a_n+b_n$ の両辺に2をかけて，
\[
2^n\alpha=2a_n+2b_n=a_{n+1}+b_{n+1} \quad \cdots \text{①}
\]
である。$n$の偶奇で場合分けして，$k\in\mathbb{N}$に対し，

\textbf{$1^\circ$ $n=2k$の時，$1<2b_n<2$だから，}
\[
\begin{cases}
a_{2k+1}=2a_{2k}+1 \\
b_{2k+1}=2b_{2k}-1
\end{cases}
\quad \cdots \text{②}
\]

\textbf{$2^\circ$ $n=2k-1$の時，$0\le2b_n<1$だから，}
\[
\begin{cases}
a_{2k}=2a_{2k-1} \\
b_{2k}=2b_{2k-1}
\end{cases}
\quad \cdots \text{③}
\]
となる。②，③から
\[
b_{2k+1}=4b_{2k-1}-1 \quad\therefore\ b_{2k+1}-\frac13=4\left(b_{2k-1}-\frac13\right)
\]
$b_1=\alpha$ だから，くり返し用いて，
\[
\begin{cases}
b_{2k-1}=4^{k-1}\left(\alpha-\dfrac13\right)+\dfrac13 \\
b_{2k}=2\cdot4^{k-1}\left(\alpha-\dfrac13\right)+\dfrac23
\end{cases}
\]
$0\le b_k<1$ が任意の$k$で成立するので，$\alpha=\dfrac13$ が必要で，逆にこの時，
\[
b_{2k-1}=\frac13,\quad b_{2k}=\frac23
\]
で条件を満たす。よって $\alpha=\dfrac13$ である。この時，$b_n$と同様に$a_n$をもとめて（$a_1=0$より），
\[
a_{2k-1}=4^{k-1}\left(0+\frac13\right)-\frac13=\frac13\left(4^{k-1}-1\right)
\]
\[
a_{2k}=\frac23\left(4^{k-1}-1\right)
\]
\end{proof}

\section*{第2問}

\begin{proof}[解]
$f(x)=x^3+3x^2$ とする。グラフの概形は右図。
\[
f'(x)=3x^2+6x
\]
から，$x=t$ での接線は
\[
y=(3t^2+6t)x-2t^3-3t^2
\]
となる。これが$O$を通る時，
\[
2t^3+3t^2=0 \iff t=0,\ -\frac32
\]
であることに注意すると，線分$OT$が$D$に含まれるような点$T$の領域は下図斜線部（境界含む）である。辺$OR$，$OS$がこの領域内にあることが必要で，逆にこの時$\triangle ORS$は$D$に含まれる。よって$R$，$S$は下図斜線部内をうごく。

\begin{center}
\begin{tikzpicture}[scale=0.85, >=stealth]
  \draw[->] (-3,0) -- (2,0) node[right] {$x$};
  \draw[->] (0,-0.8) -- (0,4.8) node[above] {$y$};
  \fill[gray!15] plot[domain=-2:1,samples=60] (\x,{\x*\x*\x+3*\x*\x}) -- (1,4) -- (-2,4) -- cycle;
  \draw[domain=-2.5:1.15, smooth, thick] plot (\x,{\x*\x*\x+3*\x*\x});
  \draw[thick] (-2,4) -- (1,4);
  \draw[thick,dashed] (-16/9,4) -- (-2.4,5.4);
  \draw[thick] (0,0) -- (-16/9,4);
  \draw[thick] (0,0) -- (1,4);
  \fill (0,0) circle (1.2pt) node[below right] {$O$};
  \fill (-2,4) circle (1.2pt) node[above left] {$P(-2,4)$};
  \fill (1,4) circle (1.2pt) node[above right] {$Q=R(1,4)$};
  \fill (-16/9,4) circle (1.2pt);
  \node[below] at (-16/9,3.75) {$S\left(-\frac{16}{9},4\right)$};
  \node[below right] at (0.3,2) {$D$};
\end{tikzpicture}
\end{center}

さて，$\triangle ORS$の$\max$を与える$R,S$の候補を考えよう。まず，$S$を固定し，$R$を$y=kx\ (k\ge0)$上で固定し，この上で動かす。$OR$を底辺とみれば，$\triangle ORS$が$\max$となるのは$OR$が最大，つまり$R$がこの領域$E$の境界上をうごく時である。

同様に考えれば，$S$も$E$の境界上の時，$\triangle ORS$は$\max$となる。そこでまず$\max$を与える$R$をもとめる。$S$を境界上で動かし，$OS: y=kx\ (k\le0)$とする。この時，$OS$と$R$の距離は$R(x_1,y_1)$として
\[
\ell=\frac{|kx_1+y_1|}{\sqrt{k^2+1}}=\frac{-kx_1+y_1}{\sqrt{k^2+1}} \quad (\because\ x_1,y_1\ge0,\ -k\ge0)
\]
これは$x_1,y_1$について単調増加だから，$\max\triangle ORS$を与える$R$の1つに$R(1,4)$がある。この時$S$と$OR$の距離を$\max$にする$S$をもとめれば良い。$S(x_2,y_2)$として，この距離を$\ell_2$とすると，$OR: y=4x$だから
\[
\ell_2=\frac{|4x_2-y_2|}{\sqrt{17}}=\frac{-4x_2+y_2}{\sqrt{17}} \quad (\because\ x_2<0,\ y_2>0)
\]
となり，同様に，これを$\max$にする$S$は$S\left(-\dfrac{16}{9},4\right)$である。

以上から，$S\left(-\dfrac{16}{9},4\right)$，$R(1,4)$の時，$\triangle ORS$は$\max$で，
\[
\frac12\left|1\cdot4+\frac{16}{9}\cdot4\right|=\frac{50}{9}
\]
となる。
\end{proof}

\section*{第3問}

\begin{proof}[解]
$P,Q$が線分$OR$上にある時を，時刻の基準として$t=0$とする。又この時の$R$の座標が$(3,0)$となるようにして考える。すると時刻$t$での各々の座標は
\[
P(\cos t,\sin t),\quad Q(2\cos2t,2\sin2t),\quad R(3\cos3t,3\sin3t)
\]
とかける。以下$S=\sin t,\ C=\cos t$とする。

\begin{center}
\begin{tikzpicture}[scale=1.1, >=stealth]
  \draw[->] (-2.3,0) -- (3.5,0) node[right] {$x$};
  \draw[->] (0,-2.3) -- (0,2.3) node[above] {$y$};
  \draw (0,0) circle (1);
  \draw (0,0) circle (2);
  \draw (0,0) circle (3);
  \fill (0,0) circle (1.2pt) node[below left] {$O$};
  \fill (0.878,0.479) circle (1.2pt) node[above] {$P$};
  \fill (1.081,1.683) circle (1.2pt) node[above] {$Q$};
  \fill (0.212,2.992) circle (1.2pt) node[above] {$R$};
  \draw[dashed] (0,0) -- (0.212,2.992);
\end{tikzpicture}
\end{center}

\[
\overrightarrow{PQ}=\begin{pmatrix}2\cos2t-C\\2\sin2t-S\end{pmatrix},\qquad
\overrightarrow{PR}=\begin{pmatrix}3\cos3t-C\\3\sin3t-S\end{pmatrix}
\]
だからサラスの公式より，時刻$t$での$\triangle PQR$の面積$T(t)$として，
\[
2T(t)=\left|(3\sin3t-S)(2\cos2t-C)-(3\cos3t-C)(2\sin2t-S)\right|
\]
\[
=\bigl|6\sin3t\cos2t-2S\cos2t-3C\sin3t+SC-(6\sin2t\cos3t-3S\cos3t-2C\sin2t+SC)\bigr|
\]
\[
=\bigl|6(\sin3t\cos2t-\sin2t\cos3t)-3(C\sin3t-S\cos3t)-2(S\cos2t-C\sin2t)\bigr|
\]
\[
=\bigl|6\sin t-3\sin2t-2\sin(-t)\bigr|
=\bigl|8S-3\sin2t\bigr|
=\bigl|8S-6SC\bigr|=|S|\,|8-6C| \quad \cdots \text{①}
\]
①の中身を$f(t)$とすると，
\[
f'(t)=8C-6\cos2t=8C-6(2C^2-1)=2(-6C^2+4C+3)
\]
から，下表をうる（$|S|,C$の周期性から，$0\le t\le\pi$で考えれば良い）。ただし，$\alpha=\dfrac{2-\sqrt{22}}{6}$である。
\begin{center}
\begin{tabular}{c|ccccc}
$t$ & $0$ & & & & $\pi$ \\
\hline
$C$ & $1$ & & $\alpha$ & & $-1$ \\
\hline
$f'$ & & $+$ & $0$ & $-$ & \\
\hline
$f$ & $0$ & $\nearrow$ & & $\searrow$ & $0$
\end{tabular}
\end{center}
ここで，$C=\alpha$の時，
\[
|S|=\sqrt{1-\alpha^2}=\frac16\sqrt{10+4\sqrt{22}}
\]
だから，①に代入して
\[
\max T(t)=\frac{\sqrt{10+4\sqrt{22}}}{6}\left|4-3\cdot\frac{2-\sqrt{22}}{6}\right|
=\frac{(6+\sqrt{22})\sqrt{10+4\sqrt{22}}}{12}
\left(=\frac{\sqrt{409+88\sqrt{22}}}{6}\right)
\]
\end{proof}

\section*{第4問}

\begin{proof}[解]
\begin{enumerate}
  \item[(1)] $t\to0$の時を考えるので，$|\cos2x|=\cos2x$として考えて良い。この時，平均値の定理から，
  \[
  F(t)=\frac{\int_0^{\frac{\pi}{2}t}\cos2x\,dx}{t}=\frac{\pi}{2}\cos2p \quad \left(0<p<\frac{\pi}{2}t\right)
  \]
  をみたす$p$がある。はさみうちから$t\to0$の時$p\to0$で，$\cos2x$は連続だから，
  \[
  F(t)\longrightarrow\frac{\pi}{2}\cos(2\cdot0)=\frac{\pi}{2} \quad (t\to0)
  \]

  \item[(2)]
  \[
  F(t)=
  \begin{cases}
  \dfrac1t\displaystyle\int_0^{\frac{\pi}{2}t}\cos2x\,dx & \left(0<t\le\dfrac12\right) \\[2mm]
  \dfrac1t\left\{\displaystyle\int_0^{\frac{\pi}{4}}\cos2x\,dx-\int_{\frac{\pi}{4}}^{\frac{\pi}{2}t}\cos2x\,dx\right\} & \left(\dfrac12\le t\le1\right)
  \end{cases}
  \]
  \[
  =
  \begin{cases}
  \dfrac{1}{2t}\sin\pi t & \left(0<t\le\dfrac12\right) \\[2mm]
  \dfrac{1}{2t}(2-\sin\pi t) & \left(\dfrac12\le t\le1\right)
  \end{cases}
  \]
  したがって，前者は$(0,0)$と$\left(t,\dfrac12\sin\pi t\right)$を結ぶ直線の傾き，後者は$(0,0)$と$\left(t,1-\dfrac12\sin\pi t\right)$を結ぶ直線の傾きを表すことから，右図とあわせて，$y=F(t)$のグラフの概形は下図。

  \begin{center}
  \begin{tikzpicture}[scale=1.15, >=stealth]
    \draw[->] (-0.2,0) -- (1.5,0) node[right] {$t$};
    \draw[->] (0,-0.3) -- (0,2) node[above] {$y$};
    \draw[dashed] (0,1) -- (1,1);
    \node[left] at (0,1) {$1$};
    \node[left] at (0,1.571) {$\frac{\pi}{2}$};
    \draw[domain=0.02:0.5, smooth, thick] plot (\x,{sin(180*\x)/(2*\x)});
    \draw[domain=0.5:1, smooth, thick] plot (\x,{(2-sin(180*\x))/(2*\x)});
    \draw[dashed] (0,0) circle (0pt);
    \fill[white] (0,1.571) circle (1.4pt);
    \draw (0,1.571) circle (1.4pt);
    \fill (0.5,1) circle (1.4pt);
    \fill (1,1) circle (1.4pt);
    \node[below] at (0.5,0) {$\frac12$};
    \node[below] at (1,0) {$1$};
  \end{tikzpicture}
  \end{center}

  したがって，$F(t)\ge1$となる$t$の範囲は，
  \[
  0<t\le\frac12,\quad t=1
  \]
\end{enumerate}
\end{proof}

\bigskip
\noindent\textbf{[解2]}

\begin{proof}[解]
$(2)$\quad $g(t)=\displaystyle\int_0^{\frac{\pi}{2}t}|\cos2x|dx$ のグラフは，$\left(\dfrac12,g\left(\dfrac12\right)\right)$を軸に対称である。$0<t\le\dfrac12$の時，
\[
g(t)=\frac12\sin\pi t
\]
であって，$y=g(t)$のグラフは右下のようになる。$F(t)$は$(0,0)$と$(t,g(t))$の傾きだから，
\[
0<t\le\frac12 \ \text{または}\ t=1
\]
\end{proof}
