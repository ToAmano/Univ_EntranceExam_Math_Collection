\begin{proof}[解]
\begin{enumerate}
  \item[(1)] 丁度1本あたるのは，$A$ から引いた球の色で場合分けして，（くじを引かない場合を含む）
  \[
  \frac{1}{6}\times\frac{{}_3C_1\cdot{}_7C_2}{{}_{10}C_3}+\frac{2}{6}\times\frac{{}_3C_1\cdot{}_7C_1}{{}_{10}C_2}+\frac{3}{6}\frac{{}_3C_1}{{}_{10}C_1}
  = \frac{1}{6}\times\frac{21}{40}+\frac{2}{6}\times\frac{14}{30}+\frac{3}{6}\times\frac{3}{10}
  = \frac{63+112+108}{720}=\frac{283}{720}
  \]

  \item[(2)] 排反を考える。1本もあたらないのは，
  \[
  \frac{1}{6}\times\frac{{}_7C_3}{{}_{10}C_3}+\frac{2}{6}\times\frac{{}_7C_2}{{}_{10}C_2}+\frac{3}{6}\frac{{}_7C_1}{{}_{10}C_1}
  = \frac{1}{6}\times\frac{7}{24}+\frac{2}{6}\times\frac{7}{15}+\frac{3}{6}\times\frac{7}{10}
  = \frac{70+224+504}{1440}=\frac{798}{1440}=\frac{133}{240}
  \]
  だから，少なくとも1本あたるのは
  \[
  1-\frac{133}{240}=\frac{107}{240}
  \]
\end{enumerate}
