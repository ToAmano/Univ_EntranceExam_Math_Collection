\section*{第1問}

\begin{proof}[解]
\[
\begin{cases}
ax+3by=1 \\
bx+ay=0
\end{cases}
\quad \cdots \text{①}
\]
を $x,y$ についてとくと，$a^2-3b^2\ne0$ のもとで，
\[
x=\frac{a}{a^2-3b^2}, \quad y=\frac{-b}{a^2-3b^2} \quad \cdots \text{②}
\]
又，$a^2-3b^2=0$ の時，$a=\pm\sqrt3 b$ であり，$b=0$ の時，$a=0$ となるが，①から $0\ne1$ で矛盾。
$b\ne0$ の時 $\pm\sqrt3=\dfrac{a}{b}$ となり，$a,b\in\mathbb{Z}$ から，(有理数)=(無理数) となり矛盾。したがって，
$a^2-3b^2\ne0$ のもとで考えれば良い。そこで，②において，$x,y\in\mathbb{Z}$ から，
\[
\begin{cases}
1^\circ\ a=0,\ b\ne0 & (\because(a,b)\ne(0,0)) \\
2^\circ\ b=0,\ a\ne0 & (\text{同上}) \\
3^\circ\ ab\ne0,\ -a\le a^2-3b^2\le a,\ -b\le a^2-3b^2\le b & (\because a,b\ge0)
\end{cases}
\]
のいずれかである。

\textbf{$1^\circ$ の時}

②から $(x,y)=\left(0,-\dfrac{1}{3b}\right)$ で，$-\dfrac{1}{3b}\in\mathbb{Z}$ とならず，（$\because b=0,1,\cdots,5$）$y\in\mathbb{Z}$ に不適。

\textbf{$2^\circ$ の時}

②から $(x,y)=\left(\dfrac{1}{a},0\right)$ で，$a=1$ のみ $x\in\mathbb{Z}$ をみたす。（$0\le a$ のもとで）よって $(a,b)=(1,0)$

\textbf{$3^\circ$ の時}

$b=0,1,\cdots,5$ で代入して，条件をみたす $a$ をあげると，（$a\ge0$）
\[
b=0 \cdots a=0, \quad b=1 \cdots a=2, \quad b=2 \cdots \text{なし}, \quad b=3 \cdots a=5, \quad b=4 \cdots a=7, \quad b=5 \cdots \text{なし}
\]
このうち，$(x,y)\in\mathbb{Z}$ とできるのは $(a,b)=(2,1)\ (7,4)$ のみである。

以上から，
\[
(a,b)=(2,1),\ (7,4),\ (1,0)
\]
\end{proof}

\section*{第2問}

\begin{proof}[解]
$f(x)=0$ の2解 $\alpha,\beta$ として，$f(x)=(x-\alpha)(x-\beta)$ とおく。
\[
f(f(x))=(f(x)-\alpha)(f(x)-\beta)=0 \quad \cdots \text{①}
\]
が重解 $r$ を持つ。$\alpha\ne\beta$ だから対称性から，$f(x)-\alpha=0$ が重解を持つ時のみ考えれば良い。$f(x)-\alpha=x^2+2x+\alpha\beta-\alpha=0$ の判別式を $D$ として $D=0$ である。
\[
\frac{D}{4}=1-(\alpha\beta-\alpha)=0 \quad\therefore\ 1+\alpha-\alpha\beta=0 \quad \cdots \text{②}
\]
$\alpha+\beta=-2$ $\therefore \beta=-(\alpha+2)$ だから，②に代入して
\[
1+\alpha+\alpha(\alpha+2)=0 \quad\Longrightarrow\quad \alpha^2+3\alpha+1=0 \quad\Longrightarrow\quad \alpha=\frac{-3\pm\sqrt5}{2}
\]
したがって，$(\alpha,\beta)=\left(\dfrac{-3\pm\sqrt5}{2},-\dfrac{1\pm\sqrt5}{2}\right)$（複号同順）となり，
\[
a=\alpha\beta=\frac{-1+\sqrt5}{2},\ \frac{-1-\sqrt5}{2}
\]
である。この時 $r=-1$ となる。
\end{proof}

\section*{第3問}

\begin{proof}[解]
与式を①とおく。真数条件から $\dfrac{x-2}{x-1}>0$ 且つ $\dfrac{3-x}{x}>0 \iff 0<x<1,\ 2<x<3\ \cdots$② である。

\textbf{$1^\circ\ 0<a<1$}

①$\iff \dfrac{x-2}{x-1}<\dfrac{3-x}{x} \iff (x-2)(x-1)x^2<(3-x)x(x-1)^2$
\[
\iff x(x-1)(2x^2-6x+3)<0 \iff 0<x<\frac{3-\sqrt3}{2},\ 1<x<\frac{3+\sqrt3}{2}
\]
だから，②とあわせて，
\[
0<x<\frac{3-\sqrt3}{2},\ 2<x<\frac{3+\sqrt3}{2}
\]

\textbf{$2^\circ\ 1<a$}

①$\iff \dfrac{x-2}{x-1}>\dfrac{3-x}{x} \iff x<0,\ \dfrac{3-\sqrt3}{2}<x<1,\ \dfrac{3+\sqrt3}{2}<x$

だから②とあわせて，
\[
\frac{3-\sqrt3}{2}<x<1,\ \frac{3+\sqrt3}{2}<x<3
\]

以上まとめて，
\[
0<a<1\ \text{の時}\quad 0<x<\frac{3-\sqrt3}{2},\ 2<x<\frac{3+\sqrt3}{2}
\]
\[
1<a\ \text{の時}\quad \frac{3-\sqrt3}{2}<x<1,\ \frac{3+\sqrt3}{2}<x<3
\]
\end{proof}

\section*{第4問}

\begin{proof}[解]
2接点を $P\left(t,\dfrac{1}{t}\right)$，$Q\left(-s,\dfrac{1}{s}\right)$ $(t,s>0)$ とすると，$P,Q$ での接線は $\left(\dfrac{1}{x}\right)'=-\dfrac{1}{x^2}$ から，
\[
\ell_P: y=-\frac{1}{t^2}x+\frac{2}{t}, \qquad \ell_Q: y=\frac{1}{s^2}x+\frac{2}{s}
\]
となり，交点 $R$，$\ell_P$ の $x$ 切片 $T$，$\ell_Q$ の $x$ 切片 $U$ とすると
\[
R\left(2\frac{ts(s-t)}{t^2+s^2},2\frac{t+s}{t^2+s^2}\right), \qquad T(2t,0), \qquad U(-2s,0)
\]

\begin{center}
\begin{tikzpicture}[scale=1.1, >=stealth]
  \draw[->] (-3,0) -- (3,0) node[right] {$x$};
  \draw[->] (0,-0.3) -- (0,3) node[above] {$y$};
  \draw[domain=0.45:2.6, smooth, thick] plot (\x, {1.6/\x});
  \draw[domain=-2.6:-0.45, smooth, thick] plot (\x, {-1.6/\x});
  \draw[thick] (-2.6,0.65) -- (2,3) node[above right] {};
  \draw[thick] (-2,3) -- (2.6,0.65);
  \fill (0.9,1.78) circle (1pt) node[above right] {$P$};
  \fill (-0.9,1.78) circle (1pt) node[above left] {$Q$};
  \fill (0,2.3) circle (1pt) node[above right] {$R$};
  \fill (1.8,0) circle (1pt) node[below] {$T$};
  \fill (-1.8,0) circle (1pt) node[below] {$U$};
  \draw[dashed] (-1.8,0) -- (0,2.3) -- (1.8,0);
\end{tikzpicture}
\end{center}

だから，題意の△の面積 $f$ は，$R$ から $x$ 軸への垂足 $H$ として
\[
f=\overline{TU}\times\overline{RH}\cdot\frac{1}{2} = \frac{1}{2}(2t+2s)\cdot2\frac{t+s}{t^2+s^2} = 2\frac{(t+s)^2}{t^2+s^2}=2\left(1+\frac{2ts}{t^2+s^2}\right) \quad \cdots \text{①}
\]
AM-GM から，$t^2+s^2\ge2ts$ $\therefore 1\ge\dfrac{2st}{t^2+s^2}$（$\because t,s>0$）で等号成立は $s=t$ の時。

よって①から，
\[
f\le2(1+1)=4
\]
となる。
\end{proof}

\section*{第5問}

\begin{proof}[解]
\begin{enumerate}
  \item[(1)] 丁度1本あたるのは，$A$ から引いた球の色で場合分けして，（くじを引かない場合を含む）
  \[
  \frac{1}{6}\times\frac{{}_3C_1\cdot{}_7C_2}{{}_{10}C_3}+\frac{2}{6}\times\frac{{}_3C_1\cdot{}_7C_1}{{}_{10}C_2}+\frac{3}{6}\frac{{}_3C_1}{{}_{10}C_1}
  = \frac{1}{6}\times\frac{21}{40}+\frac{2}{6}\times\frac{14}{30}+\frac{3}{6}\times\frac{3}{10}
  = \frac{63+112+108}{720}=\frac{283}{720}
  \]

  \item[(2)] 排反を考える。1本もあたらないのは，
  \[
  \frac{1}{6}\times\frac{{}_7C_3}{{}_{10}C_3}+\frac{2}{6}\times\frac{{}_7C_2}{{}_{10}C_2}+\frac{3}{6}\frac{{}_7C_1}{{}_{10}C_1}

  = \frac{70+224+504}{1440}=\frac{798}{1440}=\frac{133}{240}
  \]
  だから，少なくとも1本あたるのは
  \[
  1-\frac{133}{240}=\frac{107}{240}
  \]
\end{enumerate}

\end{proof}

\section*{第6問}

\begin{proof}[解]
$F(x)=\dfrac{1}{x}\displaystyle\int_0^x f(t)dt$ の両辺 $x$ で微分して
\[
F'(x)=-\frac{1}{x^2}\int_0^x f(t)dt+\frac{f(x)}{x}=\frac{1}{x}(f(x)-F(x)) \quad \cdots \text{①}
\]
である。ここで，題意から，$0\le t\le x$ の時 $f(x)\ge f(t)$ だから，同区間で積分して
\[
xf(x)\ge\int_0^x f(t)dt \quad \cdots \text{②}
\]
したがって，$0<x$ の時，②の両辺 $x$ で割って，
\[
f(x)\ge F(x) \quad \cdots \text{③}
\]
したがって，①の右辺は0以上だから，$F'(x)\ge0$。つまり $F(x)$ は単調増加。

次に，$y=G(x)=xF(x)$ とおく。
\[
G'(x)=f(x)
\]
だから，$x>0$ の時
\[
3F(x)=f(x) \iff 3xF(x)=xf(x) \iff 3G(x)=xG'(x) \quad \cdots \text{④}
\]
である。
\[
3y=x\frac{dy}{dx} \iff \frac{3}{x}dx=\frac{1}{y}dy
\]
両辺積分して，$C_0,C$ を定数として，
\[
3\log x+C_0=\log y \quad\therefore\ y=C\cdot x^3
\]
とかける。両辺微分して，$y'=f(x)$ より，
\[
f(x)=3Cx^2
\]
$f(1)=1$ から，$C=\dfrac{1}{3}$ となり，$f(x)=x^2$ である。逆にこの時，$F(x)=\dfrac{1}{3}x^2$ となり，
\begin{itemize}
\item $f(x)$ は連続かつ単調増加
\item $3F(x)=f(x)$
\item $f(1)=1$
\end{itemize}
をみたす。以上から $f(x)=x^2$

\medskip
\noindent\textbf{[解2]（後半）}

①から $f(x)=xF'(x)+F(x)$ だから，$y=F(x)$ として
\[
3F(x)=f(x) \iff 3y=x\frac{dy}{dx}+y \iff \frac{1}{x}dx=\frac{1}{2y}dy
\]
積分して，
\[
y=C\cdot x^2
\]
とおけるから，$F(1)=1$ から $C=1$ となって，$F(x)=x^2$ である。（以下略）
\end{proof}
