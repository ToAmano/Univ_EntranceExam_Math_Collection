\begin{proof}[解]
$\vec{v}=x\vec{a}+y\vec{b}+z\vec{c}$ とおく。又，与方程式 $x^2+2y^2+3z^2=1$ を①とする。

\begin{enumerate}
  \item[(1)] ①は $(x,y,z)=(1,0,0),\left(0,\dfrac{1}{\sqrt2},0\right),\left(0,0,\dfrac{1}{\sqrt3}\right)$ で成立するので，これを順に①に代入して
  \[
  |\vec{v}|=|\vec{a}|=\left|\frac{1}{\sqrt2}\vec{b}\right|=\left|\frac{1}{\sqrt3}\vec{c}\right|=1 \quad\therefore\ |\vec{a}|=1,\ |\vec{b}|=\sqrt2,\ |\vec{c}|=\sqrt3
  \]

  \item[(2)] ①は $(x,y,z)=\left(\dfrac{1}{2},\sqrt{\dfrac38},0\right),\left(\dfrac12,0,\dfrac12\right),\left(0,\dfrac12,\sqrt{\dfrac16}\right)$ で成立するので，$|\vec v|^2=1$ に代入して
  \[
  \begin{cases}
  \dfrac14|\vec a|^2+\dfrac38|\vec b|^2+2\cdot\dfrac12\sqrt{\dfrac38}\,\vec a\cdot\vec b=1 \\
  \dfrac14|\vec a|^2+\dfrac14|\vec c|^2+2\cdot\dfrac12\cdot\dfrac12\,\vec a\cdot\vec c=1 \\
  \dfrac14|\vec b|^2+\dfrac16|\vec c|^2+2\cdot\dfrac12\sqrt{\dfrac16}\,\vec b\cdot\vec c=1
  \end{cases}
  \quad\therefore\quad
  \begin{cases}
  \vec a\cdot\vec b=0 \\
  \vec a\cdot\vec c=0 \\
  \vec b\cdot\vec c=0
  \end{cases}
  \]
\end{enumerate}
\end{proof}
