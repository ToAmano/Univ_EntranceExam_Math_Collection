\subsection*{第1問}

\begin{proof}[解]
まず, $\frac{p}{q} + \frac{2}{3} \dots$ ①である. このもとで考える. 与式から
\begin{equation}
-\frac{1}{q^2} < \frac{p}{q} - \frac{2}{3} < \frac{1}{q^2} \quad \dots \text{②}
\end{equation}

まず $q > 0$ の時, ②の両辺 $q^2$ をかけて, セイリする.
\begin{equation}
\frac{2}{3}q - \frac{1}{q} < p < \frac{2}{3}q + \frac{1}{q} \quad \dots \text{③}
\end{equation}

$q = 3m \ (m \in \mathbb{N})$ の時, $|\frac{1}{q}| < 1$ から③をみたす $(p,q)$ は $(2m, 3m)$ となるが, ①からこれは不適.

次に $q = 3m+1 \ (m \in \mathbb{Z}_{\ge 0})$ の時, $\frac{2}{3}q = 2m + \frac{2}{3}$ だから右図及び $\frac{1}{q} = \frac{1}{3m+1}$ が単調減少であることから,

\begin{center}
\begin{tikzpicture}[xscale=2.0, yscale=0.8]
  \draw[->] (1.5,0) -- (3.5,0) node[right] {$p$};
  \draw (2.0,0.1) -- (2.0,-0.1) node[below] {$2m$};
  \draw (2.667,0.1) -- (2.667,-0.1) node[below] {$2m+\frac{2}{3}$};
  \draw (3.0,0.1) -- (3.0,-0.1) node[below] {$2m+1$};
  \draw[<->] (2.2,0.3) to[out=45,in=135] node[above] {$\frac{1}{q}$} (3.133,0.3);
\end{tikzpicture}
\end{center}

\begin{itemize}
  \item $\frac{2}{3} \le \frac{1}{q} \iff m=0$ の時, $p = 2m, 2m+1$
  \item $\frac{1}{3} \le \frac{1}{q} < \frac{2}{3} \iff \frac{1}{2} < m \le \frac{4}{3}$ の時, $m$ がなく不適.
  \item $\frac{1}{q} < \frac{1}{3}$ の時, ③をみたす $p$ はない.
\end{itemize}

最後に $q = 3m-1 \ (m \in \mathbb{N})$ の時, $\frac{2}{3}q = 2m - \frac{2}{3}$ だから右図より

\begin{center}
\begin{tikzpicture}[xscale=2.0, yscale=0.8]
  \draw[->] (0.5,0) -- (2.5,0) node[right] {$p$};
  \draw (1.0,0.1) -- (1.0,-0.1) node[below] {$2m-1$};
  \draw (1.333,0.1) -- (1.333,-0.1) node[below] {$2m-\frac{2}{3}$};
  \draw (2.0,0.1) -- (2.0,-0.1) node[below] {$2m$};
  \draw[<->] (0.8,0.3) to[out=45,in=135] node[above] {$\frac{1}{q}$} (1.867,0.3);
\end{tikzpicture}
\end{center}

\begin{itemize}
  \item $\frac{2}{3} \le \frac{1}{q} \iff m \le \frac{5}{6}$ の時, これをみたす $m$ はない.
  \item $\frac{1}{3} \le \frac{1}{q} < \frac{2}{3} \iff \frac{5}{6} < m \le \frac{4}{3}$ の時, $m=1$ で $p = 2m-1$.
  \item $\frac{1}{q} < \frac{1}{3}$ の時, ③をみたす $p$ はない.
\end{itemize}

以上から, $q > 0$ の時, $(p,q) = (0,1), (1,1), (1,2) \dots$ ④である.

さらに, $q < 0$ の時, ②の両辺に $q^2$ をかけて,
\[ \frac{2}{3}q + \frac{1}{q} < p < \frac{2}{3}q - \frac{1}{q} \]
$q = 3m \ (m \in \mathbb{Z})$ は $q > 0$ の時と同様不適である.

$q = -(3m+1) \ (m \in \mathbb{Z}_{\ge 0})$ の時, $\frac{2}{3}q = -(2m + \frac{2}{3})$ だから, 同様に絞ると,

\begin{center}
\begin{tikzpicture}[xscale=2.0, yscale=0.8]
  \draw[->] (-3.5,0) -- (-1.5,0) node[right] {$p$};
  \draw (-3.0,0.1) -- (-3.0,-0.1) node[below] {$-(2m+1)$};
  \draw (-2.667,0.1) -- (-2.667,-0.1) node[below] {$-(2m+\frac{2}{3})$};
  \draw (-2.0,0.1) -- (-2.0,-0.1) node[below] {$-2m$};
\end{tikzpicture}
\end{center}

$m=0$ の時 $p = -(2m+1), -2m$ である.

$q = -(3m-1) \ (m \in \mathbb{N})$ の時 $\frac{2}{3}q = -(2m - \frac{2}{3})$ で同様に $m=1$ の時 $p = -2m+1$.

以上から, $q < 0$ の時, $(p,q) = (0,-1), (-1,-1), (-1,-2) \dots$ ⑤である.

④, ⑤から
\[ (p,q) = (0, \pm 1), (\pm 1, \pm 1), (\pm 1, \pm 2) \quad (\text{複号同順}) \]
\end{proof}

\subsection*{第2問}

\begin{proof}[解]
題意をみたす整式を $f(x)$ とすると, 適当な整式 $Q(x), R(x)$ を用いて,
\begin{align}
f(x) &= Q(x) \cdot x^2 + x + 1 \quad \dots \text{①} \\
f(x) &= R(x) \cdot (x+1)^2 + x \quad \dots \text{②}
\end{align}
とかける. まず, $f(x)$ が定数又は1次の時, ①, ②を満たすものはないので以下 2次以上を考える.
①から $f(x) = \sum_{k=2}^n a_k x^k + x + 1 \ (a_k \in \mathbb{R}, a_n \neq 0, n \in \mathbb{N}_{\ge 2})$ とおける.
②から $g(x) = f(x) - x$ が $(x+1)^2$ で割りきれるので
\[ g(x) = \sum_{k=2}^n a_k x^k + 1 \]
に対し, $g(-1) = g'(-1) = 0$ であるから,
\begin{equation}
\begin{cases}
g(-1) = \sum_{k=2}^n a_k (-1)^k + 1 = 0 \\
g'(-1) = \sum_{k=2}^n k a_k (-1)^{k-1} = 0
\end{cases} \quad \dots \text{③}
\end{equation}
を満たす $a_k \ (k=2 \dots n)$ が存在する最小の $n$ をもとめれば良い.

\paragraph{$1^\circ \, n=2$}
\[ \text{③} \iff a_2 + 1 = 0 \land -2a_2 = 0 \text{ となりこれをみたす } a_2 \text{ はない.} \]

\paragraph{$2^\circ \, n=3$}
\[ \text{③} \iff \begin{cases} -a_3 + a_2 + 1 = 0 \\ 3a_3 - 2a_2 = 0 \end{cases} \iff \begin{cases} a_2 = -3 \\ a_3 = -2 \end{cases} \quad \text{だから, } n=3 \text{は適する.} \]
この時 $f(x) = -2x^3 - 3x^2 + x + 1$ となる.

以上から, もとめるのは $f(x) = -2x^3 - 3x^2 + x + 1$ である.
\end{proof}

\subsection*{第3問}

\begin{proof}[解]
\begin{align}
0 \le \alpha < \pi, \quad 0 < \beta < \frac{1}{2} &\quad \dots \text{①} \\
0 \le x \le 1, \quad y \le \cos(\pi\beta) &\quad \dots \text{②}
\end{align}
①, ②から $0 < \alpha\beta < \frac{\pi}{2}, \, 0 < \pi\beta < \frac{\pi}{2}$ だから, $\cos(\pi\beta) > 0, \, \sin(\pi\beta) > \sin(\alpha\beta) > 0$ となる.

ここで, $A = y \sin(\alpha\beta) + x \sin\{(\pi-\alpha)\beta\}$ とおく. これは ②及び $\sin(\alpha\beta) > 0$ から,
$y = \cos(\pi\beta)$ で $\max B$ となる.
\[ B = \cos(\pi\beta) \sin(\alpha\beta) + x \sin\{(\pi-\alpha)\beta\} \]
\[ \frac{dB}{d\alpha} = -\beta [ \cos(\pi\beta) \cos(\alpha\beta) + x \cos\{(\pi-\alpha)\beta\} ] \le 0 \quad (\because \text{①}, \text{②}) \]
から, $B$ は $\alpha$ の単調減少関数で, $\alpha = 0$ で最大値 $C$ をとる.
\[ C = x \sin\pi\beta \le \sin\pi\beta \quad (\text{等号成立は } x=1) \]
となる. 以上から, $A \le \sin\pi\beta$ となるから, 与式すなわち $\sin\pi\beta \le A$ が成立するのは
等号成立時のみで, この時 $x=1$ である.
\end{proof}

\subsection*{第4問}

\begin{proof}[解]
$f'(x) = 3x^2 + p, \, g'(x) = 3x^2 + 2ax$ である. $f, g$ が極値を持つので,
$f'(x) = 0, \, g'(x) = 0$ は各々 2 実解 $(\alpha, \beta), (\gamma, \delta)$ を持つ ($\alpha < \beta, \, \gamma < \delta$).
題意の条件及び 3次係数正から,
\begin{equation}
f(\alpha) = g(\gamma), \quad f(\beta) = g(\delta) \quad \dots \text{①}
\end{equation}
である. ここで, $a > 0$ 及び $g'(x) = x(3x+2a)$ から,
\begin{equation}
\gamma = -\frac{2}{3}a, \quad \delta = 0 \quad \dots \text{②}
\end{equation}
となるので, ①より,
\begin{equation}
\begin{cases}
f(\alpha) = (-\frac{2}{3}a)^3 + a(-\frac{2}{3}a)^2 + b = \frac{4}{27}a^3 + b \\
f(\beta) = b
\end{cases} \quad \dots \text{③}
\end{equation}
となる. ここで $p < 0$ のもとで $\alpha = -\sqrt{-\frac{p}{3}}, \, \beta = \sqrt{-\frac{p}{3}}$ だから, ③の辺々足し引きして,
\begin{align*}
\begin{cases}
f(\alpha) + f(\beta) = 2q = \frac{4}{27}a^3 + 2b \\
f(\alpha) - f(\beta) = -2\sqrt{-\frac{p}{3}} (\frac{2}{3}p) = \frac{4a^3}{27}
\end{cases}
\end{align*}
\[ \therefore \begin{cases} q = \frac{2}{27}a^3 + b \\ p = -\frac{a^2}{3} \end{cases} \]
\end{proof}

\subsection*{第5問}

\begin{proof}[解]
$x, y$ 軸のまわりにまわして得られる立体の体積 $V_x, V_y$ とする.

\begin{center}
\begin{tikzpicture}[scale=1.2]
  \draw[->] (-0.2,0) -- (2.2,0) node[right] {$x$};
  \draw[->] (0,-0.2) -- (0,2.2) node[above] {$y$};
  \draw[domain=0:1.5,smooth,variable=\x,thick] plot ({\x},{(1.5-\x)*(1.5-\x)});
  \node[left] at (0,2.25) {$b^2$};
  \node[below] at (1.5,0) {$b/a$};
\end{tikzpicture}
\end{center}

\[ V_x = \pi \int_0^{b/a} (ax-b)^4 dx \]
\[ V_y = 2\pi \int_0^{b/a} x(ax-b)^2 dx \]
から,
\begin{align*}
V_x = V_y &\iff \int_0^{b/a} \{ (ax-b)^4 - 2x(ax-b)^2 \} dx = 0 \\
&\iff \left[ \frac{1}{5a} (ax-b)^5 - \frac{2}{4}a^2 x^4 + \frac{4}{3}ab x^3 - b^2 x^2 \right]_0^{b/a} = 0 \\
&\iff -\frac{1}{5a} (-b)^5 - \frac{2}{4}a^2 \left(\frac{b}{a}\right)^4 + \frac{4}{3}ab \left(\frac{b}{a}\right)^3 - b^2 \left(\frac{b}{a}\right)^2 = 0 \\
&\iff \frac{b^5}{5a} - \frac{2}{4}\frac{b^4}{a^2} + \frac{4}{3}\frac{b^4}{a^2} - \frac{b^4}{a^2} = 0 \\
&\iff ab\frac{1}{5} - \frac{1}{2} + \frac{4}{3} - 1 = 0 \qquad (\because a, b \neq 0) \\
&\iff ab = \frac{5}{6} = \text{const} \quad \text{答}
\end{align*}
\end{proof}

\subsection*{第6問}

\begin{proof}[解]
$f(10) = 100a + 10b + c, \, g(10) = 100c + 10b + a$ で, $1 \le a,b,c \le 6$ から,
\begin{align*}
f(10) > 452 &\iff a=5,6 \text{ 又は } \{a=4 \land 10b+c > 52\} \\
&\iff a=5,6 \text{ 又は } \text{“}a=4 \land b=6\text{”} \text{ 又は } \text{“}a=4 \land b=5 \land c \ge 3\text{”} \\
g(10) > 452 &\iff c=5,6 \text{ 又は } \text{“}c=4 \land b=6\text{”} \text{ 又は } \text{“}c=4 \land b=5 \land a \ge 3\text{”}
\end{align*}
である. 以下のように排反に場合分けする.
\begin{enumerate}
  \item[$1^\circ$] $a=5,6$ かつ $c=5,6$
  \item[$2^\circ$] $a=5,6$ かつ $c=4$
  \item[$3^\circ$] $a=4$ かつ $c=5,6$
  \item[$4^\circ$] $a=4$ かつ $c=4$
\end{enumerate}

$1^\circ$ の時, 常に条件満たされ, 場合の数は $2 \times 6 \times 2 = 24$ 通り.

$2^\circ$ の時, $b=5,6$ なら良く, $2 \times 2 = 4$ 通り.

$3^\circ$ の時, 対称性から $2^\circ$ と同じく 4 通り.

$4^\circ$ の時, $b=5,6$ の 2 通り.

又 全事象は $6^3$ 通りで同様に確からしいので,
\[ \frac{24 + 4 + 4 + 2}{6^3} = \frac{17}{108} \quad \text{答} \]
\end{proof}
