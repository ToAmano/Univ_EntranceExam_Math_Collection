\section*{第1問}

\begin{proof}[解]
$a,b\in\mathbb{Z}_{>0}\ \cdots$①

\begin{enumerate}
  \item[(1)] $c=a+b,\ d=a^2-ab+b^2$，$d=(a-b)^2+ab>0\ (\because①)$から，
  \[
  c^2-d=3ab>0 \qquad 4d-c^2=3(a-b)^2\ge0 \qquad \cdots *
  \]
  から
  \[
  1<\frac{c^2}{d}\le4
  \]

  \item[(2)] $A=a^3+b^3$とおく。$A=p^n$（$p\in\text{prime}$，$n\in\mathbb{Z}$）となる条件をしらべる。まず①から$A\in\mathbb{Z}_{\ge2}$なので，$n\in\mathbb{N}$として良い。さらに$A=cd$から，$c=p^\alpha,\ d=p^\beta$（$\alpha,\beta\in\mathbb{Z}_{\ge0}$）とおける。(1)から
  \[
  1<p^{2\alpha-\beta}\le4 \quad\cdots\text{②}
  \]
  $p\in\text{prime}$から，②がみたされる時，
  \[
  (p,2\alpha-\beta)=(2,2)\ \cdots\text{⑦} \qquad (3,1)\ \cdots\text{⑦}
  \]
  である（$p=2,\ 2\alpha-\beta=1$の場合は$4d-c^2=2^\beta\cdot2/3\not\in\mathbb{Z}$となり不適）。

  \textbf{⑦の時}
  \[
  a+b=2^\alpha, \qquad a^2-ab+b^2=2^{2\alpha-2}
  \]
  まず，$*$で等号が成立するので，$a=b$である。代入して
  \[
  a=2^{\alpha-1}
  \]
  ①とあわせて，$(a,b)=(2^n,2^n)\ (n\in\mathbb{Z}_{\ge0})$

  \textbf{⑦の時}
  \[
  a+b=3^\alpha, \qquad a^2-ab+b^2=3^{2\alpha-1} \quad\cdots\text{③}
  \]
  $d$を消す
  \[
  3a^2-3ab+3b^2=a^2+2ab+b^2
  \]
  \[
  (2a-b)(a-2b)=0
  \]
  対称性から，$b=2a$の時のみかんがえる。③に代入
  \[
  a=3^{\alpha-1}
  \]
  したがって，$(a,b)=(3^n,2\cdot3^n)\ (n\in\mathbb{Z}_{\ge0})$。対称性から
  \[
  (a,b)=(2\cdot3^n,3^n)
  \]
  も解。

  以上から
  \[
  (a,b)=(2^n,2^n),\ (2\cdot3^n,3^n),\ (3^n,2\cdot3^n)\quad(n\in\mathbb{Z}_{\ge0})
  \]
\end{enumerate}
\end{proof}

\section*{第2問}

\begin{proof}[解]
$n\in\mathbb{Z}_{\ge3}\ \cdots$①，$x,y,z\in\mathbb{N}\ \cdots$②
\[
\begin{cases}
x+y+z=n \\
x\le y+z \\
y\le z+x \\
z\le x+y
\end{cases}
\quad\cdots *_1
\]
まず，$z=k$と固定する。$*_1$に代入して
\[
\begin{cases}
x+y=n-k\ge k \\
y\ge x-k \\
y\le x+k
\end{cases}
\quad\cdots *_2
\]
$y=n-k-x$を代入，$y>0$から$x<n-k\ \cdots$③
\[
x-k\le n-k-x\le x+k
\]
\[
\frac{n}{2}-k\le x\le\frac{n}{2}\quad\cdots\text{④}
\]
又，$1\le k\le\dfrac{n}{2}$である。同区間では$\dfrac{n}{2}\le n-k$である $\cdots$⑤

\textbf{⑦ $n=2p\ (p\in\mathbb{N})$の時（$p=2,3,\cdots$）}

$z=k$と固定（$k=1,2,\cdots,p$）した時の$(x,y)$の数は，④⑤より
\[
\begin{cases}
k=1,2,\cdots,p-1\text{の時} & k+1\text{個} \\
k=p\text{の時} & p-1\text{個}
\end{cases}
\]
より，あわせて
\[
\sum_{k=1}^{p-1}(k+1)+(p-1)=\left(\frac{p(p+1)}{2}-1\right)+(p-1)=\frac{1}{2}p^2+\frac{3}{2}p-2=\frac18n^2+\frac34n-2\ \text{個}
\]

\textbf{⑦ $n=2p+1$の時}

$z=k$と固定（$k=1,2,\cdots,p$）した時の$(x,y)$の数は，③④より$k$個で，
\[
\sum_{k=1}^{p}k=\frac12p(p+1)=\frac12\cdot\frac{n-1}{2}\cdot\frac{n+1}{2}\ \text{個}
\]

したがって，
\[
\begin{cases}
\dfrac18n^2+\dfrac34n-2\text{個} & (n:\text{even}) \\[2mm]
\dfrac12\cdot\dfrac{n-1}{2}\cdot\dfrac{n+1}{2}\text{個} & (n:\text{odd})
\end{cases}
\]
\end{proof}

\section*{第3問}

\begin{proof}[解]
$P_1\left(t_1,\dfrac1{t_1}\right),\ P_2\left(t_2,-\dfrac1{t_2}\right)$とおく（$t_1>0,\ t_2<0$）。

\begin{center}
\begin{tikzpicture}[scale=1.0, >=stealth]
  \draw[->] (-3.2,0) -- (3.2,0) node[right] {$x$};
  \draw[->] (0,-0.4) -- (0,3) node[above] {$y$};
  \draw[domain=0.35:3, smooth] plot (\x,{1/\x}) node[right] {$y=\frac1x\ (C_1)$};
  \draw[domain=-3:-0.35, smooth] plot (\x,{-1/\x}) node[left] {$y=-\frac1x\ (C_2)$};
  \fill (1.6,0.625) circle (1.2pt) node[above] {$P_1$};
  \fill (-0.663,1.509) circle (1.2pt) node[above] {$P_2$};
  \draw[domain=-1.2:2.8] plot (\x,{-\x/2.56+1.25});
  \fill (0,0) circle (1.2pt) node[below left] {$O$};
\end{tikzpicture}
\end{center}

\begin{enumerate}
  \item[(1)] 対称性から，$P_1$で$\ell$と$C_1$が接する場合を考える。この時，
  \[
  \ell: y=-\frac{1}{t_1^2}x+\frac{2}{t_1}
  \]
  であるから，$P_2\left((1-\sqrt2)t_1,\ \frac{1}{(\sqrt2-1)t_1}\right)$となり，
  \[
  \triangle OP_1P_2=\frac12\left|(1-\sqrt2)t_1\cdot\frac{1}{t_1}-\frac{1}{(\sqrt2-1)t_1}\cdot t_1\right|=\frac12\left|(1-\sqrt2)-(\sqrt2+1)\right|=\sqrt2=\text{const.}
  \]
  よって示された。

  \item[(2)] 一般の場合，$P=\dfrac{t_1}{t_2}$とおくと，
  \[
  \triangle OP_1P_2=\frac12\left|\frac{t_1}{t_2}+\frac{t_2}{t_1}\right|=\frac12\left|P+\frac1P\right|=-\frac12\left(P+\frac1P\right)
  \]
  （$t_1>0,\ t_2<0$より，$P$は$P<0$なる任意の実数をとる。）
  \[
  \triangle OP_1P_2=\sqrt2 \iff P^2+2\sqrt2P+1=0 \quad\therefore\ P=-\sqrt2\pm1
  \]

  \textbf{$1^\circ$ $P=-\sqrt2+1$の時}

  $t_1=(1-\sqrt2)t_2$とおけるが，この時，
  \[
  \ell: y=\frac{1}{t_2^2}x-\frac{2}{t_2}
  \]
  となり，$\ell$は$C_2$の接線。

  \textbf{$2^\circ$ $P=-\sqrt2-1$の時}

  $t_1=-(1+\sqrt2)t_2$とおけるが，この時，
  \[
  \ell: y=-\frac{x}{t_1^2}+\frac{2}{t_1}
  \]
  となり，$\ell$は$C_1$の接線。

  以上から示された。
\end{enumerate}
\end{proof}

\section*{第4問}

\begin{proof}[解]
$F=\displaystyle\int_0^1e^x|x-a|dx$とおく。

\textbf{⑦ $a\le0$の時}
\[
F=-a\int_0^1e^xdx+\int_0^1xe^xdx, \qquad \frac{dF}{da}=-\int_0^1e^xdx<0
\]
より，$F$は$a$の単調減少関数

\textbf{⑦ $a\ge1$の時}

⑦と同じく，$F$は$a$の単調増加関数

従って，$0\le a\le1$の時$F$は最小値をとる。$f(x)=e^x(x-a)$とおく。
\[
F=-\int_0^af(x)dx+\int_a^1f(x)dx
\]
$g(x)=e^x(x-a-1)$とおくと$g'(x)=f(x)$であって，
\[
F=g(1)+g(0)-2g(a)
\]
\[
=-ea-a-1+2e^a
\]
\[
\frac{dF}{da}=-e-1+2e^a
\]
より，下表を得る。
\begin{center}
\begin{tabular}{c|ccccc}
$a$ & $0$ & & $\log\dfrac{e+1}{2}$ & & $1$ \\
\hline
$F'$ & & $-$ & $0$ & $+$ & \\
\hline
$F$ & & $\searrow$ & & $\nearrow$ &
\end{tabular}
\end{center}
したがって，$a=\log\dfrac{e+1}{2}$で$\min F$をとる。
\end{proof}

\section*{第5問}

\begin{proof}[解]
$S=\sin x,\ C=\cos x$と書く。
\[
\tan x=\cos x \quad\therefore\ S=C^2=1-S^2 \quad\therefore\ S=\frac12(-1\pm\sqrt5)
\]
$|S|\le1$から$S=\dfrac{-1+\sqrt5}{2}\equiv S_0$である。 $\cdots$①

グラフは下図で，交点$A$の$x$座標を$\alpha$とおく。$(\tan\alpha)'=\dfrac{1}{C^2}$だから，①とあわせて，$\ell$の傾きは
\[
\frac{1}{1-S_0^2}=\frac{1}{S_0}=\frac{2}{\sqrt5-1}=\frac{\sqrt5+1}{2}
\]
であって，$A$から$x$軸に下ろした垂足$B$，$\ell$と$x$軸の交点$C$とすると，もとめる面積$T$として
\[
T=\int_0^\alpha\tan x\,dx-\triangle ABC \quad\cdots\text{①}
\]

\begin{center}
\begin{tikzpicture}[scale=1.6, >=stealth]
  \draw[->] (-0.9,0) -- (1.3,0) node[right] {$x$};
  \draw[->] (0,-0.3) -- (0,1.7) node[above] {$y$};
  \begin{scope}
    \clip (-0.9,-0.3) rectangle (1.3,1.7);
    \draw[domain=-0.9:1.0, smooth] plot (\x,{tan(\x r)}) node[right] {$y=\tan x$};
    \draw[domain=-0.9:0.9, smooth] plot (\x,{cos(\x r)}) node[right] {$y=\cos x$};
  \end{scope}
  \draw[domain=0:1.0, thick] plot (\x,{0.786+1.618*(\x-0.666)});
  \fill (0.666,0.786) circle (0.8pt) node[above] {$A$};
  \fill (0.666,0) circle (0.8pt) node[below] {$B$};
  \fill (0.180,0) circle (0.8pt) node[below] {$C$};
  \node at (0.35,1.35) {$\ell$};
\end{tikzpicture}
\end{center}

まず，
\[
\triangle ABC=\frac12\left(\sqrt{\frac{\sqrt5-1}{2}}\right)^2\cdot\frac{1}{\frac{\sqrt5+1}{2}}=\frac12\cdot\frac{\sqrt5-1}{\sqrt5+1}=\frac12\cdot\frac{(\sqrt5-1)^2}{4}\quad\cdots\text{②}
\]
さらに
\[
\int_0^\alpha\tan x\,dx=\left[-\log(\cos x)\right]_0^\alpha=-\log(\cos\alpha)=-\frac12\log\frac{\sqrt5-1}{2}\quad\cdots\text{③}
\]
①，②，③から
\[
T=-\frac12\log\frac{\sqrt5-1}{2}-\frac18(\sqrt5-1)^2
\]
\end{proof}
