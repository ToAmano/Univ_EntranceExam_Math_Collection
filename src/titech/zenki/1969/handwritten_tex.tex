\section*{第1問}

[解]
\[
\begin{cases}
a^2 - b^2 - c^2 > 0 & \dots \text{①} \\
ap < 0, \quad x > 0 & \dots \text{②} \\
ax + by + cz = p & \dots \text{③}
\end{cases}
\]

$yz$平面で③の表す直線を $l$ と考える。まず、$(b, c) = (0, 0)$ とすると、$ax = p$ となるが、これは②に反する。よって $(b, c) \neq (0, 0)$ である。このもとで
\[
\min (y^2 + z^2) = \left( \frac{|p - ax|}{\sqrt{b^2 + c^2}} \right)^2 = \frac{(p - ax)^2}{b^2 + c^2}
\]
だから、
\begin{align*}
x^2 - (y^2 + z^2) &\le x^2 - \frac{(p - ax)^2}{b^2 + c^2} < x^2 - \frac{(p - ax)^2}{a^2} \quad (\because \text{①}) \\
&= + 2 \frac{p}{a} x - \left( \frac{p}{a} \right)^2 = \left( \frac{p}{a} \right) \left( 2x - \frac{p}{a} \right) < 0 \quad (\because \text{②より}, \frac{p}{a} < 0, x > 0)
\end{align*}
となり、
\[
x^2 - y^2 - z^2 < 0_{\mathbin{/\mkern-6mu/}}
\]
である。

\section*{第3問}

[解] $\alpha = x + y, \beta = xy$ とすると、存在条件 ($x, y$ の実数条件) から $\alpha^2 - 4\beta \ge 0 \quad \dots \text{①}$ である。
\[
u = \alpha + 1, \quad v = 1 - 2\beta \quad \dots \text{②}
\]
であり、$x^2 + y^2 = a^2$ から
\[
\alpha^2 - 2\beta = a^2 \quad \dots \text{③}
\]
②から $\alpha = u - 1, \beta = \frac{1}{2}(1 - v)$ だから、①, ③に代入して
\[
\begin{cases}
(u-1)^2 - 2(1-v) \ge 0 \\
(u-1)^2 - (1-v) = a^2
\end{cases}
\iff
\begin{cases}
v \ge \frac{1}{2}(-u^2 + 2u + 1) \\
v = -u^2 + 2u + a^2
\end{cases}
\quad \text{\dots (1)}
\]

\noindent
(2) $\frac{1}{\sqrt{2}} \le a \le 1$ の時、$\frac{1}{2} \le a^2 \le 1$ だから、(1)から
\[
\begin{cases}
-u^2 + 2u + \frac{1}{2} \le v \le -u^2 + 2u + 1 \\
v \ge \frac{1}{2}(-u^2 + 2u + 1)
\end{cases}
\]

\section*{第4問}

[解] $3a_n > 2a_{n-1} \quad \dots \text{①}$

与式から
\[
\begin{pmatrix} X \\ Y \end{pmatrix} = x \begin{pmatrix} a_n \\ a_{n-1} \end{pmatrix} + y \begin{pmatrix} 2 \\ 3 \end{pmatrix}
\]
である。$\vec{a}_n = \begin{pmatrix} a_n \\ a_{n-1} \end{pmatrix}$, $\vec{\beta} = \begin{pmatrix} 2 \\ 3 \end{pmatrix}$ とおくと、$|x| + |y| \le 1$ から、$(X, Y)$ は右下斜線部内をうごく ($O$ に注意した)。この面積 $S$ は、
\begin{align*}
\frac{1}{2} S &= (\text{$\vec{\beta}$ と $-\vec{a}_n$ でつくられる三角形}) + (\text{$\vec{\beta}$ と $\vec{a}_n$ でつくられる三角形}) \\
&= \frac{1}{2} |-3a_n + 2a_{n-1}| + \frac{1}{2} |3a_n - 2a_{n-1}| \\
&= |3a_n - 2a_{n-1}| \\
&= 3a_n - 2a_{n-1} \quad (\because \text{①})
\end{align*}
と表せる。$S = 2$ だから、
\[
1 = 3a_n - 2a_{n-1} \quad (n \ge 2)
\]
\[
\therefore a_n - 1 = \frac{2}{3}(a_{n-1} - 1) \quad (n \ge 2)
\]
くり返し用いて、
\[
a_n = \left(\frac{2}{3}\right)^{n-1} (a_1 - 1) + 1 \longrightarrow 1 \quad (n \to +\infty)_{\mathbin{/\mkern-6mu/}}
\]

\section*{第5問}

[解] $f_n(x) = \sum_{k=0}^n a_k x^k - x$ とおくと、$f'(x) = \sum_{k=1}^n k a_k x^{k-1} - 1$ であり、$f(0) = a_0 > 0$,
$f(1) = 0 \quad (\because \sum_{k=0}^n a_k = 1)$。又 $a_k \ge 0$ から $f'(x)$ は $0 \le x \le 1$ で単調増加で、グラフの概形は右のようになる。

\subsection*{$1^\circ$ 十分性}
$f(x) = 0$ が、$0 < x < 1$ に唯一つの実根 $\alpha$ を持つとする。平均値の定理から
\[
\frac{f(1) - f(\alpha)}{1 - \alpha} = f'(c) \quad (\alpha < c < 1)
\]
\[
\therefore 0 = f'(c)
\]
となる $c$ がある。$f''(x) \ge 0$ から
\[
0 < f'(1) = \sum_{k=1}^n k a_k - 1 \quad \therefore 1 < \sum_{k=1}^n k a_k
\]

\subsection*{$2^\circ$ 必要性}
$\sum_{k=1}^n k a_k - 1 > 0 \implies f'(1) > 0$ の時、$1$ より小さく $1$ に十分近い $\alpha$ に対して $f(\alpha) < 0$ となる
($\because f'(x) \ge 0$) から、$f(x)$ の連続性および単調性から、$0 < x < 1$ にぴったり実解を持つ。

\section*{第6問}

[解] $t = \frac{x}{y}$ とおくと、$0 < t < 1 \quad \dots \text{①}$ である。$x^2 e^{\frac{y}{x}} > 0, y^2 e^{\frac{x}{y}} > 0$ に注意する。
\[
f(t) = \frac{x^2 e^{\frac{y}{x}}}{y^2 e^{\frac{x}{y}}} = t^2 e^{\frac{1}{t} - t}
\]
\begin{align*}
f'(t) &= e^{\frac{1}{t} - t} \left( t^2 \left(-1 - \frac{1}{t^2}\right) + 2t \right) \\
&= e^{\frac{1}{t} - t} (-t^2 + 2t - 1) \\
&= -e^{\frac{1}{t} - t} (t - 1)^2 < 0
\end{align*}
から、$0 < t \le 1$ で $f(t)$ は単調減少。これと、$f(1) = 1$ から、$f(t) > 1$ なので
\[
x^2 e^{\frac{y}{x}} > y^2 e^{\frac{x}{y}}_{\mathbin{/\mkern-6mu/}}
\]
である。
