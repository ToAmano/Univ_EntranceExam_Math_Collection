自然数$n$に対して，
\begin{align*}
  a_n=\int_0^1{(1+x)}^{-n-1}e^{x^2}dx
\end{align*}
\begin{align*}
  b_n=\int_0^1{(1+x)}^{-n}xe^{x^2}dx
\end{align*}
とおく．
\begin{enumerate}
  \item $\displaystyle b_n \leqq e\cdot\int_0^1{(1+x)}^{-n}dx$が成り立つことを示し，
$\displaystyle\lim_{n\to\infty}b_n$を求めよ．
  \item $\displaystyle\lim_{n\to\infty}na_n$を求めよ．
\end{enumerate}
