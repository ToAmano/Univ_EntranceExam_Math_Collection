[\textbf{解}] $P$ での接線は $\frac{x_1}{a^2}x - \frac{y_1}{b^2}y = 1 \dots \text{①}$ であるから，$Q\left(\frac{a^2}{x_1}, 0\right)$ である．

\begin{enumerate}
    \item[(1)] まず $\frac{y_1^2}{b^2} = \frac{x_1^2}{a^2} - 1$ 及び $y_1, b > 0$ から
    \[
    y_1 = b \sqrt{\frac{x_1^2}{a^2} - 1}
    \]
    だから
    \[
    \triangle OPQ = \frac{1}{2} \frac{a^2}{x_1} b \sqrt{\frac{x_1^2}{a^2} - 1}
    \]

    \item[(2)]
    \[
    \begin{aligned}
    \triangle OPQ &= \frac{ab}{2} \sqrt{\frac{x_1^2 - a^2}{x_1^2}} \\
    &= \frac{ab}{2} \sqrt{1 - a^2 / x_1^2} \xrightarrow{x_1 \to \infty} \frac{ab}{2}
    \end{aligned}
    \]
\end{enumerate}
