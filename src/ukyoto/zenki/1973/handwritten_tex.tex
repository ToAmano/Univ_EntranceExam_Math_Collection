\subsection*{各問のメモ・解答概要}
\begin{enumerate}
    \item 3桁の整数の百の位と一の位を入れ替えた差 $\alpha - \alpha' = 99(A-C)$。その差の値の種類（19個、正のものは9個）、および正の値の総和 ($S = 4455$)。
    \item 3次方程式 $x^3 + px^2 + qx + r = 0$ の解の絶対値がすべて1で1以外の2根を $\alpha, \beta$ とするときの実数係数 $(p, q, r)$ の条件。$(p = -q, r = -1, -1 \le q \le 3)$ または $(p, q, r) = (-1, -1, 1)$。
    \item 三角形 $ABC$ の頂点 $C$ の位置ベクトル $\vec{c}$ を $\vec{a}, \vec{b}$ で表す。$\vec{c} = \frac{b}{a}\vec{a} + \frac{a}{b}\vec{b}$。

    \item 関数の増減と極値の評価による値域 $1 \le f(x) < 2$ の証明。
    \item 期待値 $E(a), E(b), E(c)$ の大きさを $p$ の範囲で比較・分類。
\end{enumerate}

\section*{第1問}

\noindent
[\textbf{解}]
\begin{enumerate}
    \item[(1)] $\alpha = 100A + 10B + C$ ($A, B, C$ は $0 \sim 9$ までの整数) とおける。
    この時、$\alpha' = 100C + 10B + A$ だから
    \begin{align*}
    \alpha - \alpha' &= 100(A - C) + (C - A) \\
    &= 99(A - C)
    \end{align*}
    ここに $A, C$ は $0$ 以上 $9$ 以下の整数だから、$A - C$ は $-9$ 以上 $9$ 以下の全ての整数をとるので、$A - C$ のとり得る値の個数は $9 - (-9) + 1 = \underline{19\text{個}}$。このうち正のものは $A - C$ が $1 \sim 9$ のものだから、$\beta$ は \underline{9個}。

    \item[(2)] (1)から総和 $S$ として
    \[
    S = \sum_{k=1}^9 99k = \frac{99}{2} \cdot 9 \cdot 10 = \underline{4455}
    \]
\end{enumerate}

\section*{第2問}

\noindent
[\textbf{解}] $e(\theta) = \cos\theta + i\sin\theta$ とする。1以外の2根を $\alpha, \beta$ とすると、
\[
\alpha = e(\theta_1), \quad \beta = e(\theta_2) \qquad (0 \le \theta_1 \le \theta_2 < 2\pi)
\]
とおける。解と係数の関係から、
\begin{align*}
p &= -(1 + e(\theta_1) + e(\theta_2)) \quad \dots \text{①} \\
q &= e(\theta_1) + e(\theta_2) + e(\theta_1)e(\theta_2) \quad \dots \text{②} \\
r &= -e(\theta_1)e(\theta_2) = -e(\theta_1 + \theta_2) \quad \dots \text{③}
\end{align*}

\begin{enumerate}
    \item[$1^\circ$] $\alpha, \beta \in \mathbb{R}$ の時\\
    $(\alpha, \beta) = (1, 1), (1, -1), (-1, -1)$ で、①〜③より、
    \[
    (p, q, r) = (1, -1, -1), (-1, -1, 1), (-3, 3, -1)
    \]

    \item[$2^\circ$] $\alpha, \beta \notin \mathbb{R}$ の時\\
    $p, q, r \in \mathbb{R}$ から $\beta = \bar{\alpha}$ となり、$0 \le \theta_1 \le \theta_2 < 2\pi$ とあわせて、$\theta_2 = 2\pi - \theta_1$, $0 < \theta_1 < \pi$ とできる。
    この時
    \[
    \begin{cases}
    r = -e(2\pi) = -1 \\
    p = -(1 + 2\cos\theta_1) \\
    q = 1 + 2\cos\theta_1
    \end{cases}
    \]
    だから、
    \[
    p = -q, \quad r = -1, \quad -1 \le q \le 3
    \]
\end{enumerate}
以上から
\[
(p = -q, \ r = -1, \ -1 \le q \le 3) \quad \text{or} \quad (p, q, r) = (-1, -1, 1)
\]

\section*{第3問}

\noindent
[\textbf{解1}] $O$ を原点とし、$B$ が $x$ 軸上、$A$ が第一象限になるよう $xy$ 平面をおく。
この時 $|\vec{a}| = a$, $|\vec{b}| = b$ とすると、$\angle AOB = \frac{\pi}{3}$ より、
\[
\vec{b} = b \begin{pmatrix} 1 \\ 0 \end{pmatrix}, \quad \vec{a} = \frac{a}{2} \begin{pmatrix} 1 \\ \sqrt{3} \end{pmatrix}
\]
とおける。$AB$ の中点を $M$ とする。
\[
\vec{AB} = \begin{pmatrix} b - \frac{1}{2}a \\ -\frac{\sqrt{3}}{2}a \end{pmatrix}
\]
だから、$|MC| = \frac{\sqrt{3}}{2}|AB|$, $MC \perp AB$, $C$ が $O$ と反対側であることをあわせて、
\[
\vec{MC} = \frac{\sqrt{3}}{2} \begin{pmatrix} \frac{\sqrt{3}}{2}a \\ b - \frac{1}{2}a \end{pmatrix}
\]
となり、
\begin{align*}
\vec{c} &= \vec{OM} + \vec{MC} \\
&= a \begin{pmatrix} 1 \\ 0 \end{pmatrix} + \frac{1}{2}b \begin{pmatrix} 1 \\ \sqrt{3} \end{pmatrix} \quad \dots \text{①}
\end{align*}
一方、
\[
\frac{b}{a}\vec{a} + \frac{a}{b}\vec{b} = b \cdot \frac{1}{2} \begin{pmatrix} 1 \\ \sqrt{3} \end{pmatrix} + a \begin{pmatrix} 1 \\ 0 \end{pmatrix} \quad \dots \text{②}
\]
①, ②から
\[
\vec{c} = \frac{b}{a}\vec{a} + \frac{a}{b}\vec{b}
\]
となる。

\noindent
[\textbf{解2}] $\vec{a}, \vec{b}$ に平行な単位ベクトルを $\vec{x}, \vec{y}$ とすると、これらは1次独立で、又 $\vec{x} \cdot \vec{y} = \frac{1}{2}$ となる。さらに
\[
\vec{c} = \alpha \vec{x} + \beta \vec{y}
\]
とおける。
\begin{align*}
\vec{AB} &= |\vec{b}|\vec{y} - |\vec{a}|\vec{x} \\
\vec{AC} &= (\alpha - |\vec{a}|)\vec{x} + \beta \vec{y} \\
\vec{BC} &= \alpha \vec{x} + (\beta - |\vec{b}|)\vec{y}
\end{align*}
これらの大きさ2乗が全て等しいので、$k \ge 0$ として
\[
\begin{cases}
|\vec{b}|^2 + |\vec{a}|^2 - |\vec{a}||\vec{b}| = k \\
(\alpha - |\vec{a}|)^2 + \beta^2 + \beta(\alpha - |\vec{a}|) = k \\
\alpha^2 + (\beta - |\vec{b}|)^2 + \alpha(\beta - |\vec{b}|) = k
\end{cases}
\quad \dots (*)
\]
$(\alpha, \beta) = (|\vec{b}|, |\vec{a}|)$ とすると、(*)は成立する。又、$\vec{x}, \vec{y}$ は1次独立であることから、コレ以外の解には他ならず、
\begin{align*}
\vec{c} &= |\vec{b}|\vec{x} + |\vec{a}|\vec{y} \\
&= |\vec{b}|\frac{\vec{a}}{|\vec{a}|} + |\vec{a}|\frac{\vec{b}}{|\vec{b}|}
\end{align*}
である。

\section*{第4問}

\noindent
[\textbf{解}] $a = \sqrt{n}$, $b = \sqrt{n+1}$ とする。
\begin{align*}
a_n &= n^2 \left[ \frac{(\sqrt{n(n+1)} - n)^3}{n} - \frac{(\sqrt{n(n+1)} - (n+1))^3}{n+1} \right] \\
&= a^2 \left[ \frac{(ab - a^2)^3}{a^2} - \frac{(ab - b^2)^3}{b^2} \right] \\
&= a^2 \left[ a(b - a)^3 - b(a - b)^3 \right] \\
&= a^2 (a + b)(b - a)^3 \\
&= a^2 (b - a)^2 \qquad (\because (a + b)(b - a) = b^2 - a^2 = 1) \\
&= \frac{a^2}{(a + b)^2} (b - a)^2 (b + a)^2 \\
&= \frac{a^2}{(a + b)^2} \\
&= \frac{1}{\left(1 + \sqrt{1 + \frac{1}{n}}\right)^2} \xrightarrow{n \to \infty} \underline{\frac{1}{4}}
\end{align*}

\section*{第5問}

\noindent
[\textbf{解}]
\[
f'(x) = x + \frac{x^2}{2} + \dots + \frac{x^n}{n}
\]
\[
f''(x) = 1 + x + \dots + x^{n-1}
\]
であり、$x = 1$ の時 $f''(x) = n > 0$、$x \neq 1$ の時 $f''(x) = \frac{1 - x^n}{1 - x}$ だから $-1 \le x \le 1$ では $f''(x) > 0$、つまり $f'(x)$ は単調増加。これと $f'(0) = 0$ から下表をえる。

\begin{center}
\begin{tabular}{c|ccccc}
$x$ & $-1$ & $\dots$ & $0$ & $\dots$ & $1$ \\ \hline
$f'$ & & $-$ & $0$ & $+$ & \\ \hline
$f$ & & $\searrow$ & $1$ & $\nearrow$ &
\end{tabular}
\end{center}

従って、$-1 \le x \le 1$ では $1 \le f(x) \le \max\{ f(1), f(-1) \}$ $\dots$ ① である。

\begin{align*}
f(1) &= 1 + \frac{1}{1 \cdot 2} + \dots + \frac{1}{n(n+1)} \\
&= 1 + \sum_{k=1}^n \left( \frac{1}{k} - \frac{1}{k+1} \right) \\
&= 2 - \frac{1}{n+1} < 2 \quad (\because n > 0)
\end{align*}

\[
f(-1) = 1 + \frac{1}{1 \cdot 2} - \frac{1}{2 \cdot 3} + \dots + \frac{(-1)^{n+1}}{n(n+1)}
\]
より $n=1$ の時 $f(-1) = \frac{3}{2}$、$n \ge 2$ の時
\[
f(-1) = f(1) - 2 \left( \frac{1}{2 \cdot 3} + \frac{1}{4 \cdot 5} + \dots \right) < f(1)
\]
だから①より、
\[
1 \le f(x) < 2
\]

[\textbf{解}] Aの期待値は $1$ $\dots$ ①

\noindent
Bの期待値 $E(b)$ は、1回目があたるかどうかで場合分けして
\[
E(b) = 2p + p(1-p) = -p^2 + 3p \quad \dots \text{②}

\noindent
Cの期待値 $E(c)$ も同様に
\[
E(c) = 3p^2 + 2p(1-p) = p^2 + 2p \quad \dots \text{③}

$0 < p < 1$ と右のグラフから
\[
\begin{cases}
0 < p < \frac{3 - \sqrt{5}}{2} \text{の時} & \text{A} \\

\frac{3 - \sqrt{5}}{2} < p < \frac{1}{2} \text{の時} & \text{B} \\
p = \frac{1}{2} \text{の時} & \text{B, C} \\
