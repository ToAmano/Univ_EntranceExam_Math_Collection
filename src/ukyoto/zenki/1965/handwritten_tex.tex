\section*{第2問}

\noindent
[\textbf{解}] $k > 0 \dots \text{①}$。根の1つを $\alpha$ とすると、$k$ についての恒等式
\[
(\alpha - a)k^2 + (\alpha^2 - 4\alpha + 4)k + b - 4\alpha = 0
\]
が成り立つから
\[
a = \alpha, \quad (\alpha - 2)^2 = 0, \quad b = 4\alpha
\]
\[
\Rightarrow \begin{cases} \alpha = 2 \\ a = 2 \\ b = 8 \end{cases} \quad \text{--- (H)}
\]
である。代入して与方程式は
\[
k x^2 - (k+2)^2 x + (2k^2 + 4k + 8) = 0
\]
\[
(x - 2)\left(k x - (k^2 + 2k + 4)\right) = 0
\]
だから、もう片方の解は
\[
x = \frac{k^2 + 2k + 4}{k} = k + \frac{4}{k} + 2 \ge 8
\]
($\text{∵ } k > 0$ から AM-GM, 等号成立は $k = \frac{4}{k} \ \therefore k = 2$)

\noindent
だから、$k = 2$ の時 $\min 8$ をとる。

\section*{第3問}

\noindent
[\textbf{解}]
\begin{align}
&-\frac{\pi}{2} \le \alpha, \beta \le \frac{\pi}{2} \dots \text{①} \\
&\sin(x+\alpha) + \sin(x+\beta) = \sqrt{3}\sin x \dots \text{②}
\end{align}
②が $x$ についての恒等式なので、$x = 0, \frac{\pi}{2}$ での成立が必要で
\begin{align}
\sin\alpha + \sin\beta &= 0 \dots \text{③} \\
\cos\alpha + \cos\beta &= \sqrt{3} \dots \text{④}
\end{align}
①, ③から $\beta = -\alpha \dots \text{⑤}$ である。④に代入して
\[
2\cos\alpha = \sqrt{3} \iff \cos\alpha = \frac{\sqrt{3}}{2} \iff \alpha = \pm\frac{\pi}{6} \quad (\text{∵ } \text{①})
\]
だから、
\[
(\alpha, \beta) = \left( \pm\frac{\pi}{6}, \mp\frac{\pi}{6} \right) \quad (\text{複号同順})
\]
が必要。逆にこの時、
\[
(\text{与左辺}) = 2\sin x \cos\frac{\pi}{6} = \sqrt{3}\sin x = (\text{右辺})
\]
となって分（十分）。よって
\[
(\alpha, \beta) = \left( \pm\frac{\pi}{6}, \mp\frac{\pi}{6} \right) \quad (\text{複号同順})
\]

だから $t_1, t_2$ は方程式
\begin{equation}
\frac{|\beta(t-\alpha) + \alpha\beta|}{-\alpha(t-\alpha) + \beta^2} = \frac{|\beta(t\cos\theta-\alpha) - \alpha(t\sin\theta-\beta)|}{-\alpha(t\cos\theta-\alpha) - \beta(t\sin\theta-\beta)} \dots \text{①}
\end{equation}
の2異実数解（ただし $C = \cos\theta, S = \sin\theta$ ときて、$\begin{pmatrix} \alpha \\ \beta \end{pmatrix} = r \begin{pmatrix} \cos\varphi \\ \sin\varphi \end{pmatrix}$ ($0 < r, 0 < \varphi < \theta$) とおいて、①に代入すると
\[
\frac{|r\sin\varphi \cdot t|}{-r\cos\varphi \cdot t + r^2} = \frac{|r(\sin\varphi \cdot C - \cos\varphi \cdot S) t|}{-r(\cos\varphi \cdot C + \sin\varphi \cdot S) t + r^2}

\[
\frac{\sin\varphi}{-\cos\varphi \cdot t + r} = \frac{\sin(\theta-\varphi)}{-\cos(\theta-\varphi) \cdot t + r}
\]
\[
\left(\sin\varphi \cos(\theta-\varphi) - \sin(\theta-\varphi)\cos\varphi\right) t + r\left(\sin\varphi - \sin(\theta-\varphi)\right) = 0
\]
\begin{equation}
\sin(2\varphi - \theta) t + r\left(\sin\varphi - \sin(\theta-\varphi)\right) = 0 \dots \text{②}
\end{equation}
②が2異実数解を持つ条件は、②の左辺が $t$ の1次式であることと $t > 0$ より
\begin{align}
\sin(\theta - 2\varphi) &= 0 \dots \text{③} \\
\sin\varphi &= \sin(\theta - \varphi) \dots \text{④}
\end{align}
$0 < \theta < \pi, 0 < \varphi < \theta$ から ③より
\[

\]
が必要。この時④もみたされ、十分。だから題意のようになる斜線は

\section*{第5問}

\noindent
[\textbf{解}] $g(x) = \dfrac{f(x) - f(a)}{x - a}$ とおく。\\
$\displaystyle \lim_{x \to a+0} g(x)$ と $\displaystyle \lim_{x \to a-0} g(x)$ が一致する時、$\displaystyle \lim_{x \to a} g(x)$ を $f(x)$ の $x = a$ における微分係数という。

\paragraph{(ア)} $f(x) = \dfrac{1}{x^3}$, $g(x) = \dfrac{\frac{1}{x^3} - 1}{x - 1}$ の時、
\[
g(x) = \frac{1 - x^3}{x^3} \frac{1}{x - 1} = \frac{-1}{x^3}(x^2 + x + 1)
\]
より、
\[
\lim_{x \to 1+0} g(x) = \lim_{x \to 1-0} g(x) = -3 \quad \mathbin{/\mkern-5mu/}
\]

\paragraph{(イ)} $f(x) = \sqrt{x^2 + x + 1}$, $g(x) = \dfrac{\sqrt{x^2 + x + 1} - \sqrt{3}}{x - 1}$ の時、
\begin{align*}
g(x) &= \frac{1}{x - 1} \frac{(x^2 + x - 2)}{\sqrt{x^2 + x + 1} + \sqrt{3}} \\
&= \frac{x + 2}{\sqrt{x^2 + x + 1} + \sqrt{3}}
\end{align*}
から
\[
\lim_{x \to 1+0} g(x) = \lim_{x \to 1-0} g(x) = \frac{3}{2\sqrt{3}} = \frac{1}{2}\sqrt{3} \quad \mathbin{/\mkern-5mu/}
\]

\section*{第6問}

\noindent
[\textbf{解}]
\[
\int_1^x (x - t) f(x) \, dt = x^4 - 2x^2 + 1 \dots \text{①}
\]
\begin{align*}
(\text{左辺}) &= x f(x) \int_1^x dt - f(x) \int_1^x t \, dt \\
&= x f(x) (x - 1) - f(x) \frac{1}{2} (x^2 - 1) \\
&= \frac{1}{2} f(x) (x - 1)^2
\end{align*}
\[
(\text{右辺}) = (x + 1)^2 (x - 1)^2
\]
だから、①が $1 < x$ の全ての $x$ で成立する時
\[
f(x) = 2(x + 1)^2
\]
であるから、$f(x) = 2(x + 1)^2$ \hfill $\mathbin{/\mkern-5mu/}$
