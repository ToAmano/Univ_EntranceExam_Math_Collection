  A_1=\begin{pmatrix}\\
    2 \\1
  \end{pmatrix},
  A_2=\begin{pmatrix}\\
    3 \\1
  \end{pmatrix},
  A_{n+2}=\begin{pmatrix}\\
    1 & 1 \\ 1 & -1 \\
  \end{pmatrix}A_{n+1}+
  \begin{pmatrix}\\
    0 & -1 \\ 1 & 0 \\
  \end{pmatrix}A_{n},
  (n=1,2,3,\cdots)
\end{equation}
