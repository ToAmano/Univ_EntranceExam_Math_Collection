\begin{enumerate}
\item底辺の長さが$l$，2つの底角が$\alpha$，$\beta$の三角形の面積$S$は、次式で与えられることを示せ．
\[ S=\frac{l^2}{4}\cdot\frac{\cos(\alpha-\beta)-\cos(\alpha+\beta)}{\sin(\alpha+\beta)} \]
  \item各辺の長さが1，2，$\sqrt3$の三角形の各辺に1点ずつ頂点をもつ正三角形の面積の最小値を求めよ．
\end{enumerate}
