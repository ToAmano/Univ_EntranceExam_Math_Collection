\begin{enumerate}
\item 極限値$\displaystyle \lim_{n \to \infty} \sum_{k=n}^{2n} \frac{1}{k}$を求めよ．
  \item 任意の正数$a$に対して$\displaystyle \lim_{n \to \infty} \sum_{k=n}^{2n} \frac{1}{a+k}$は
(1)と同じ極限値をもつことを証明せよ．
\end{enumerate}
