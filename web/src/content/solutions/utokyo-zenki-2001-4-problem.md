複素数平面上の点$a_1,a_2,\cdots,a_n,\cdots$を
$\begin{cases}
a_1=1,  a_2=i, \\
a_{n+2}=a_{n+1}+a_n  (n=1,2,\cdots)
\end{cases}$
により定め，
$\displaystyle b_n=\frac{a_{n+1}}{a_n}$ $(n=1,2,\cdots)$とおく．
ただし，$i$は虚数単位である．

\begin{enumerate}
\item3点$b_1$，$b_2$，$b_3$を通る円$C$の中心と半径を求めよ．
  \itemすべての点$b_n$ $(n=1,2,\cdots)$は円$C$の周上にあることを示せ．
\end{enumerate}
