\begin{enumerate}
\item自然数$n=1,2,3,\cdots\cdots$に対して，
ある多項式$p_n(x)$，$q_n(x)$が存在して，
$\sin n\theta=p_n(\tan\theta)\cos^n\theta$，$\cos n\theta=q_n(\tan\theta)\cos^n\theta$と書けることを示せ．
  \itemこのとき，$n>1$ならば次の等式が成立することを証明せよ．
\[ {p_n}'(x)=nq_{n-1}(x), {q_n}'(x)=-np_{n-1}(x) \]
\end{enumerate}
