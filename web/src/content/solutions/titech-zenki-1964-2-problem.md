$a, \, b, \, c, \, d$は正数とする。

\begin{enumerate}
\item 次の2つの不等式を証明せよ。

$$
\begin{align*}
\frac{a+b+c}{3}\geqq\sqrt[3]{abc}\, ; \frac{a+b+c+d}{4}\geqq\sqrt[4]{abcd}
\end{align*}
$$

  \item 次の式で与えられる$P, \, Q, \, R, \, S$の大小を比較せよ。

$$
\begin{align*}
P=\frac{a+b+c+d}{4},  Q=\sqrt[4]{abcd},
\end{align*}
$$

$$
\begin{align*}
R=\frac{\sqrt{ab}+\sqrt{ac}+\sqrt{ad}+\sqrt{bc}+\sqrt{bd}+\sqrt{cd}}{6},
\end{align*}
$$

$$
\begin{align*}
S=\frac{\sqrt[3]{abc}+\sqrt[3]{abd}+\sqrt[3]{acd}+\sqrt[3]{bcd}}{4}
\end{align*}
$$

\end{enumerate}
