\begin{enumerate}
\item $x>0$のとき，不等式$e^{-x}>1-x$が成り立つことを証明せよ．
  \item 次の極限値を求めよ．

$$
\begin{align*}
\lim_{n\to\infty}\left(1-\frac{1}{1+\sqrt2}\right)\left(1-\frac{1}{\sqrt2+\sqrt3}\right)\cdots\left(1-\frac{1}{\sqrt{n-1}+\sqrt{n}}\right)
\end{align*}
$$

\end{enumerate}
